29 while ( size > 0 ) {

99static esp_err_t
100http_file_handler(httpd_req_t *req)
101{
102 // The header is at the start of the output file, and says where everything else is.
103 // The filesystem entries are at the end of the filesystem, after all of the file
104 // names and file data, because their table can
105 // only be written after all files have been processed. File names and file data
106 // are interleaved after the header. Position of things does matter somewhat because
107 // it's a block-based FLASH device.
108 const struct compressed_fs_header * const header = (const struct compressed_fs_header *)fs;
109 struct compressed_fs_entry * entries = (struct compressed_fs_entry *)(fs + header->table_offset);
110 gm_uri uri = {};
111
112 if ( gm_uri_parse(req->uri, &uri) != 0 )
113 return ESP_ERR_INVALID_ARG;
114
115 // Really simple sequential search for now.
116 // There would have to be lots of files for this to be a problem.
117 for ( int i = 0; i < header->number_of_files; i++ ) {
118 const char * const name = (fs + entries[i].name_offset);
119 if ( strcmp(name, uri.path) == 0 ) {
120 // The file was found. Serve it.
121 read_file(req, fs, &entries[i]);
122 return ESP_OK;
123 }
124 }
125 // If we get here, the file was not found.
126 if ( gm_web_handler_run(req, &uri, GET) == 0 ) {
127 // We found a handler for this URL. Return OK.
128 return ESP_OK;
129 }
130 else {
131 httpd_resp_send_404(req);
132 return ESP_OK;
133 }
134}
