113 :param dkw: keyword arguments passed to the Retrying object
114 """
115 # support both @retry and @retry() as valid syntax
116 if len(dargs) == 1 and callable(dargs[0]):
117 return retry()(dargs[0])

120 def wrap(f: WrappedFn) -> WrappedFn:
121 if isinstance(f, retry_base):
122 warnings.warn(
123 f"Got retry_base instance ({f.__class__.__name__}) as callable argument, "
124 f"this will probably hang indefinitely (did you mean retry={f.__class__.__name__}(...)?)"
125 )
126 if iscoroutinefunction(f):
127 r: "BaseRetrying" = AsyncRetrying(*dargs, **dkw)
128 elif tornado and hasattr(tornado.gen, "is_coroutine_function") and tornado.gen.is_coroutine_function(f):

292 @property
293 def statistics(self) -> t.Dict[str, t.Any]:
294 """Return a dictionary of runtime statistics.
295
296 This dictionary will be empty when the controller has never been
297 ran. When it is running or has ran previously it should have (but
298 may not) have useful and/or informational keys and values when
299 running is underway and/or completed.
300
301 .. warning:: The keys in this dictionary **should** be some what
302 stable (not changing), but there existence **may**
303 change between major releases as new statistics are
304 gathered or removed so before accessing keys ensure that
305 they actually exist and handle when they do not.
306
307 .. note:: The values in this dictionary are local to the thread
308 running call (so if multiple threads share the same retrying
309 object - either directly or indirectly) they will each have
310 there own view of statistics they have collected (in the
311 future we may provide a way to aggregate the various
312 statistics from each thread).
313 """

315 return self._local.statistics
316 except AttributeError:
317 self._local.statistics = {}

338 def begin(self) -> None:
339 self.statistics.clear()
340 self.statistics["start_time"] = time.monotonic()
341 self.statistics["attempt_number"] = 1
342 self.statistics["idle_for"] = 0
343
344 def iter(self, retry_state: "RetryCallState") -> t.Union[DoAttempt, DoSleep, t.Any]: # noqa
345 fut = retry_state.outcome
346 if fut is None:
347 if self.before is not None:
348 self.before(retry_state)
349 return DoAttempt()
350
351 is_explicit_retry = retry_state.outcome.failed and isinstance(retry_state.outcome.exception(), TryAgain)
352 if not (is_explicit_retry or self.retry(retry_state=retry_state)):
353 return fut.result()
354
355 if self.after is not None:
356 self.after(retry_state)
357
358 self.statistics["delay_since_first_attempt"] = retry_state.seconds_since_start
359 if self.stop(retry_state=retry_state):
360 if self.retry_error_callback:
361 return self.retry_error_callback(retry_state)
362 retry_exc = self.retry_error_cls(fut)
363 if self.reraise:
364 raise retry_exc.reraise()
365 raise retry_exc from fut.exception()
366
367 if self.wait:
368 sleep = self.wait(retry_state=retry_state)
369 else:
370 sleep = 0.0
371 retry_state.next_action = RetryAction(sleep)
372 retry_state.idle_for += sleep
373 self.statistics["idle_for"] += sleep
374 self.statistics["attempt_number"] += 1
375
376 if self.before_sleep is not None:
377 self.before_sleep(retry_state)
378
379 return DoSleep(sleep)
380
381 def __iter__(self) -> t.Generator[AttemptManager, None, None]:
382 self.begin()
383
384 retry_state = RetryCallState(self, fn=None, args=(), kwargs={})
385 while True:
386 do = self.iter(retry_state=retry_state)
387 if isinstance(do, DoAttempt):
388 yield AttemptManager(retry_state=retry_state)
389 elif isinstance(do, DoSleep):
390 retry_state.prepare_for_next_attempt()
391 self.sleep(do)
392 else:
393 break

400class Retrying(BaseRetrying):
401 """Retrying controller."""
402
403 def __call__(self, fn: t.Callable[..., _RetValT], *args: t.Any, **kwargs: t.Any) -> _RetValT:
404 self.begin()
405
406 retry_state = RetryCallState(retry_object=self, fn=fn, args=args, kwargs=kwargs)
407 while True:
408 do = self.iter(retry_state=retry_state)
409 if isinstance(do, DoAttempt):
410 try:
411 result = fn(*args, **kwargs)
412 except BaseException: # noqa: B902
413 retry_state.set_exception(sys.exc_info())
414 else:
415 retry_state.set_result(result)
416 elif isinstance(do, DoSleep):
417 retry_state.prepare_for_next_attempt()
418 self.sleep(do)
419 else:
420 return do

446class RetryCallState:
447 """State related to a single call wrapped with Retrying."""
448
449 def __init__(
450 self,
451 retry_object: BaseRetrying,
452 fn: t.Optional[WrappedFn],
453 args: t.Any,
454 kwargs: t.Any,
455 ) -> None:
456 #: Retry call start timestamp
457 self.start_time = time.monotonic()
458 #: Retry manager object
459 self.retry_object = retry_object
460 #: Function wrapped by this retry call
461 self.fn = fn
462 #: Arguments of the function wrapped by this retry call
463 self.args = args
464 #: Keyword arguments of the function wrapped by this retry call
465 self.kwargs = kwargs
466
467 #: The number of the current attempt
468 self.attempt_number: int = 1
469 #: Last outcome (result or exception) produced by the function
470 self.outcome: t.Optional[Future] = None
471 #: Timestamp of the last outcome
472 self.outcome_timestamp: t.Optional[float] = None
473 #: Time spent sleeping in retries
474 self.idle_for: float = 0.0
475 #: Next action as decided by the retry manager
476 self.next_action: t.Optional[RetryAction] = None
477
478 @property
479 def seconds_since_start(self) -> t.Optional[float]:
480 if self.outcome_timestamp is None:
481 return None
482 return self.outcome_timestamp - self.start_time
483
484 def prepare_for_next_attempt(self) -> None:
485 self.outcome = None
486 self.outcome_timestamp = None
487 self.attempt_number += 1
488 self.next_action = None
489
490 def set_result(self, val: t.Any) -> None:
491 ts = time.monotonic()
492 fut = Future(self.attempt_number)
493 fut.set_result(val)
494 self.outcome, self.outcome_timestamp = fut, ts
495
496 def set_exception(self, exc_info: t.Tuple[t.Type[BaseException], BaseException, "types.TracebackType"]) -> None:
497 ts = time.monotonic()
498 fut = Future(self.attempt_number)
499 fut.set_exception(exc_info[1])
500 self.outcome, self.outcome_timestamp = fut, ts
501
502 def __repr__(self):
503 if self.outcome is None:
504 result = "none yet"
505 elif self.outcome.failed:
506 exception = self.outcome.exception()
507 result = f"failed ({exception.__class__.__name__} {exception})"
508 else:
509 result = f"returned {self.outcome.result()}"
510
511 slept = float(round(self.idle_for, 2))
512 clsname = self.__class__.__name__
513 return f"<{clsname} {id(self)}: attempt #{self.attempt_number}; slept for {slept}; last result: {result}>"
