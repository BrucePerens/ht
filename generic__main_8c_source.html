89 // Create and store an AES key used for encrypting cookie data for

91 // storing session data on the host, but this is a memory-constrained

93 // The esp_fill_random() operation has to happen after WiFi is enabled,

107 GM.nvs,
108 iv_name,

122 || hmac_key_size != sizeof(GM.aes_cookie_iv)

151 sizeof(GM.aes_cookie_iv));
152

154
155 esp_err_t err;
156 if ( (err = set_key_err)
157 || (err = set_iv_err)
158 || (err = set_hmac_key_err)
159 || (err = nvs_commit_error) ) {

163 }
