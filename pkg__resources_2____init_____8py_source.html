447def compatible_platforms(provided, required):
448 """Can code for the `provided` platform run on the `required` platform?
449
450 Returns true if either platform is ``None``, or the platforms are equal.
451
452 XXX Needs compatibility checks for Linux and other unixy OSes.
453 """
454 if provided is None or required is None or provided == required:
455 # easy case
456 return True
457
458 # macOS special cases
459 reqMac = macosVersionString.match(required)
460 if reqMac:
461 provMac = macosVersionString.match(provided)
462
463 # is this a Mac package?
464 if not provMac:
465 # this is backwards compatibility for packages built before
466 # setuptools 0.6. All packages built after this point will
467 # use the new macOS designation.
468 provDarwin = darwinVersionString.match(provided)
469 if provDarwin:
470 dversion = int(provDarwin.group(1))
471 macosversion = "%s.%s" % (reqMac.group(1), reqMac.group(2))
472 if (
473 dversion == 7
474 and macosversion >= "10.3"
475 or dversion == 8
476 and macosversion >= "10.4"
477 ):
478 return True
479 # egg isn't macOS or legacy darwin
480 return False
481
482 # are they the same major version and machine type?
483 if provMac.group(1) != reqMac.group(1) or provMac.group(3) != reqMac.group(3):
484 return False
485
486 # is the required OS major update >= the provided one?
487 if int(provMac.group(2)) > int(reqMac.group(2)):
488 return False
489
490 return True
491
492 # XXX Linux and other platforms' special cases should go here
493 return False

759 def resolve(
760 self,
761 requirements,
762 env=None,
763 installer=None,
764 replace_conflicting=False,
765 extras=None,
766 ):
767 """List all distributions needed to (recursively) meet `requirements`
768
769 `requirements` must be a sequence of ``Requirement`` objects. `env`,
770 if supplied, should be an ``Environment`` instance. If
771 not supplied, it defaults to all distributions available within any
772 entry or distribution in the working set. `installer`, if supplied,
773 will be invoked with each requirement that cannot be met by an
774 already-installed distribution; it should return a ``Distribution`` or
775 ``None``.
776
777 Unless `replace_conflicting=True`, raises a VersionConflict exception
778 if
779 any requirements are found on the path that have the correct name but
780 the wrong version. Otherwise, if an `installer` is supplied it will be
781 invoked to obtain the correct version of the requirement and activate
782 it.
783
784 `extras` is a list of the extras to be used with these requirements.
785 This is important because extra requirements may look like `my_req;
786 extra = "my_extra"`, which would otherwise be interpreted as a purely
787 optional requirement. Instead, we want to be able to assert that these
788 requirements are truly required.
789 """
790
791 # set up the stack
792 requirements = list(requirements)[::-1]
793 # set of processed requirements
794 processed = {}
795 # key -> dist
796 best = {}
797 to_activate = []
798
799 req_extras = _ReqExtras()
800
801 # Mapping of requirement to set of distributions that required it;
802 # useful for reporting info about conflicts.
803 required_by = collections.defaultdict(set)
804
805 while requirements:
806 # process dependencies breadth-first
807 req = requirements.pop(0)
808 if req in processed:
809 # Ignore cyclic or redundant dependencies
810 continue
811
812 if not req_extras.markers_pass(req, extras):
813 continue
814
815 dist = self._resolve_dist(
816 req, best, replace_conflicting, env, installer, required_by, to_activate
817 )
818
819 # push the new requirements onto the stack
820 new_requirements = dist.requires(req.extras)[::-1]
821 requirements.extend(new_requirements)
822
823 # Register the new requirements needed by req
824 for new_requirement in new_requirements:
825 required_by[new_requirement].add(req.project_name)
826 req_extras[new_requirement] = req.extras
827
828 processed[req] = True
829
830 # return list of distros to activate
831 return to_activate

833 def _resolve_dist(
834 self, req, best, replace_conflicting, env, installer, required_by, to_activate
835 ):
836 dist = best.get(req.key)
837 if dist is None:
838 # Find the best distribution and add it to the map
839 dist = self.by_key.get(req.key)

864 def find_plugins(self, plugin_env, full_env=None, installer=None, fallback=True):
865 """Find all activatable distributions in `plugin_env`
866
867 Example usage::
868
869 distributions, errors = working_set.find_plugins(
870 Environment(plugin_dirlist)
871 )
872 # add plugins+libs to sys.path
873 map(working_set.add, distributions)
874 # display errors
875 print('Could not load', errors)
876
877 The `plugin_env` should be an ``Environment`` instance that contains
878 only distributions that are in the project's "plugin directory" or
879 directories. The `full_env`, if supplied, should be an ``Environment``
880 contains all currently-available distributions. If `full_env` is not
881 supplied, one is created automatically from the ``WorkingSet`` this
882 method is called on, which will typically mean that every directory on
883 ``sys.path`` will be scanned for distributions.
884
885 `installer` is a standard installer callback as used by the
886 ``resolve()`` method. The `fallback` flag indicates whether we should
887 attempt to resolve older versions of a plugin if the newest version
888 cannot be resolved.
889
890 This method returns a 2-tuple: (`distributions`, `error_info`), where
891 `distributions` is a list of the distributions found in `plugin_env`
892 that were loadable, along with any other distributions that are needed
893 to resolve their dependencies. `error_info` is a dictionary mapping
894 unloadable plugin distributions to an exception instance describing the
895 error that occurred. Usually this will be a ``DistributionNotFound`` or
896 ``VersionConflict`` instance.
897 """
898
899 plugin_projects = list(plugin_env)
900 # scan project names in alphabetic order
901 plugin_projects.sort()
902
903 error_info = {}
904 distributions = {}
905
906 if full_env is None:
907 env = Environment(self.entries)
908 env += plugin_env
909 else:
910 env = full_env + plugin_env
911
912 shadow_set = self.__class__([])
913 # put all our entries in shadow_set
914 list(map(shadow_set.add, self))
915
916 for project_name in plugin_projects:
917
918 for dist in plugin_env[project_name]:
919
920 req = [dist.as_requirement()]
921
922 try:
923 resolvees = shadow_set.resolve(req, env, installer)
924
925 except ResolutionError as v:
926 # save error info
927 error_info[dist] = v
928 if fallback:
929 # try the next older version of project
930 continue
931 else:
932 # give up on this project, keep going
933 break
934
935 else:
936 list(map(shadow_set.add, resolvees))
937 distributions.update(dict.fromkeys(resolvees))
938
939 # success, no need to try any more versions of this project
940 break
941
942 distributions = list(distributions)
943 distributions.sort()
944
945 return distributions, error_info

2889 insert just ahead of the parent,
2890 removing any lower-priority entries.
2891 - Else: add it to the front of path.
