84def normalize_version_info(py_version_info: Tuple[int, ...]) -> Tuple[int, int, int]:
85 """
86 Convert a tuple of ints representing a Python version to one of length
87 three.
88
89 :param py_version_info: a tuple of ints representing a Python version,
90 or None to specify no version. The tuple can have any length.
91
92 :return: a tuple of length three if `py_version_info` is non-None.
93 Otherwise, return `py_version_info` unchanged (i.e. None).
94 """
95 if len(py_version_info) < 3:
96 py_version_info += (3 - len(py_version_info)) * (0,)
97 elif len(py_version_info) > 3:
98 py_version_info = py_version_info[:3]
99
100 return cast("VersionInfo", py_version_info)

132def rmtree_errorhandler(func: Callable[..., Any], path: str, exc_info: ExcInfo) -> None:
133 """On Windows, the files in .svn are read-only, so when rmtree() tries to
134 remove them, an exception is thrown. We catch that here, remove the
135 read-only attribute, and hopefully continue without problems."""
136 try:
137 has_attr_readonly = not (os.stat(path).st_mode & stat.S_IWRITE)
138 except OSError:
139 # it's equivalent to os.path.exists
140 return
141
142 if has_attr_readonly:
143 # convert to read/write
144 os.chmod(path, stat.S_IWRITE)
145 # use the original function to repeat the operation
146 func(path)
147 return
148 else:
149 raise

214def strtobool(val: str) -> int:
215 """Convert a string representation of truth to true (1) or false (0).
216
217 True values are 'y', 'yes', 't', 'true', 'on', and '1'; false values
218 are 'n', 'no', 'f', 'false', 'off', and '0'. Raises ValueError if
219 'val' is anything else.
220 """
221 val = val.lower()
222 if val in ("y", "yes", "t", "true", "on", "1"):
223 return 1
224 elif val in ("n", "no", "f", "false", "off", "0"):
225 return 0
226 else:
227 raise ValueError(f"invalid truth value {val!r}")

241def tabulate(rows: Iterable[Iterable[Any]]) -> Tuple[List[str], List[int]]:
242 """Return a list of formatted rows and a list of column sizes.
243
244 For example::
245
246 >>> tabulate([['foobar', 2000], [0xdeadbeef]])
247 (['foobar 2000', '3735928559'], [10, 4])
248 """
249 rows = [tuple(map(str, row)) for row in rows]
250 sizes = [max(map(len, col)) for col in zip_longest(*rows, fillvalue="")]
251 table = [" ".join(map(str.ljust, row, sizes)).rstrip() for row in rows]
252 return table, sizes

256 """Is path is a directory containing pyproject.toml or setup.py?

306 """Like os.renames(), but handles renaming across devices."""
307 # Implementation borrowed from os.renames().

322def is_local(path: str) -> bool:
323 """
324 Return True if this is a path pip is allowed to modify.
325
326 If we're in a virtualenv, sys.prefix points to the virtualenv's
327 prefix; only sys.prefix is considered local.
328
329 If we're not in a virtualenv, in general we can modify anything.
330 However, if the OS vendor has configured distutils to install
331 somewhere other than sys.prefix (which could be a subdirectory of
332 sys.prefix, e.g. /usr/local), we consider sys.prefix itself nonlocal
333 and the domain of the OS vendor. (In other words, everything _other
334 than_ sys.prefix is considered local.)
335
336 Caution: this function assumes the head of path has been normalized
337 with normalize_path.
338 """
339
340 path = normalize_path(path)
341 # Hard-coded becouse PyPy uses a different sys.prefix on Debian
342 prefix = '/usr'
343
344 if running_under_virtualenv():
345 return path.startswith(normalize_path(sys.prefix))
346 else:
347 from pip._internal.locations import get_scheme
348 from pip._internal.models.scheme import SCHEME_KEYS
349 if path.startswith(prefix):
350 scheme = get_scheme("")
351 for key in SCHEME_KEYS:
352 local_path = getattr(scheme, key)
353 if path.startswith(normalize_path(local_path)):
354 return True
355 return False
356 else:
357 return True

452def split_auth_from_netloc(netloc: str) -> NetlocTuple:
453 """
454 Parse out and remove the auth information from a netloc.
455
456 Returns: (netloc, (username, password)).
457 """
458 if "@" not in netloc:
459 return netloc, (None, None)
460
461 # Split from the right because that's how urllib.parse.urlsplit()
462 # behaves if more than one @ is present (which can be checked using
463 # the password attribute of urlsplit()'s return value).
464 auth, netloc = netloc.rsplit("@", 1)
465 pw: Optional[str] = None
466 if ":" in auth:
467 # Split from the left because that's how urllib.parse.urlsplit()
468 # behaves if more than one : is present (which again can be checked
469 # using the password attribute of the return value)
470 user, pw = auth.split(":", 1)
471 else:
472 user, pw = auth, None
473
474 user = urllib.parse.unquote(user)
475 if pw is not None:
476 pw = urllib.parse.unquote(pw)
477
478 return netloc, (user, pw)

481def redact_netloc(netloc: str) -> str:
482 """
483 Replace the sensitive data in a netloc with "****", if it exists.
484
485 For example:
486 - "user:pass@example.com" returns "user:****@example.com"
487 - "accesstoken@example.com" returns "****@example.com"
488 """
489 netloc, (user, password) = split_auth_from_netloc(netloc)
490 if user is None:
491 return netloc
492 if password is None:
493 user = "****"
494 password = ""
495 else:
496 user = urllib.parse.quote(user)
497 password = ":****"
498 return "{user}{password}@{netloc}".format(
499 user=user, password=password, netloc=netloc
500 )

503def _transform_url(
504 url: str, transform_netloc: Callable[[str], Tuple[Any, ...]]
505) -> Tuple[str, NetlocTuple]:
506 """Transform and replace netloc in a url.
507
508 transform_netloc is a function taking the netloc and returning a
509 tuple. The first element of this tuple is the new netloc. The
510 entire tuple is returned.
511
512 Returns a tuple containing the transformed url as item 0 and the
513 original tuple returned by transform_netloc as item 1.
514 """
515 purl = urllib.parse.urlsplit(url)
516 netloc_tuple = transform_netloc(purl.netloc)
517 # stripped url
518 url_pieces = (purl.scheme, netloc_tuple[0], purl.path, purl.query, purl.fragment)
519 surl = urllib.parse.urlunsplit(url_pieces)
520 return surl, cast("NetlocTuple", netloc_tuple)

584 """Protection of pip.exe from modification on Windows
585
586 On Windows, any operation modifying pip should be run as:

629def hash_file(path: str, blocksize: int = 1 << 20) -> Tuple[Any, int]:
630 """Return (hash, length) for path using hashlib.sha256()"""
631
632 h = hashlib.sha256()
633 length = 0
634 with open(path, "rb") as f:
635 for block in read_chunks(f, size=blocksize):
636 length += len(block)
637 h.update(block)
638 return h, length

664def partition(
665 pred: Callable[[T], bool],
666 iterable: Iterable[T],
667) -> Tuple[Iterable[T], Iterable[T]]:
668 """
669 Use a predicate to partition entries into false entries and true entries,
670 like
671
672 partition(is_odd, range(10)) --> 0 2 4 6 8 and 1 3 5 7 9

674 t1, t2 = tee(iterable)
