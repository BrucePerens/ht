79 int32_t max_out_len = ascii85_get_max_encoded_length(in_len);
80 LCUT_TRUE(tc, max_out_len > 0);
81
82 uint8_t obuf[max_out_len + 1];
83
84 int32_t olen = encode_ascii85(tv->in, in_len, obuf, max_out_len);
85
86 LCUT_TRUE(tc, olen >= 0);
87 LCUT_TRUE(tc, olen <= max_out_len);
88
89 obuf[olen] = 0u;
90
91 LCUT_TRUE(tc, (0 == strcmp((char *)obuf, (const char *)tv->out)));

208static uint32_t random_size (void)
209{
210 uint32_t x = (xorshift128plus_next() % 8u) + 1u;
211 uint32_t y = (xorshift128plus_next() % 8u) + 1u;
212 uint32_t size = xorshift128plus_next() & ((1u << ((x / y) + 4)) - 1u);
213
214 if (size < 1u)
215 {
216 //fprintf(stderr, "size increased from %u to 1\n", size);
217 size = 1u;
218 }
219 else if (size > MAX_A85_SIZE)
220 {
221 //fprintf(stderr, "size decreased from %u to %u\n", size, MAX_A85_SIZE);
222 size = MAX_A85_SIZE;
223 }
224 else
225 {
226 // ok
227 }
228
229 return (size);
230}

234static void tc_a85_random (lcut_tc_t *tc, void *data)
235{
236 uint8_t ibuf[MAX_A85_SIZE];
237 uint8_t obuf[MAX_A85_SIZE + (MAX_A85_SIZE / 2u)];
238 uint8_t dbuf[MAX_A85_SIZE * 6u];
239
240 int count = 100000;
241
242 uint64_t rand = 0u;
243
244 (void )data;
245
246 if (rand_seed == 0)
247 {
248 // Spread out the changing bits since only a few are likely to change between runs
249 rand_seed = murmurhash3_avalanche(time(NULL));
250 rand_seed ^= murmurhash3_avalanche((uint64_t )&rand_seed);
251 }
252
253 xorshift128plus_seed(rand_seed);
254
255 while (count--)
256 {
257 uint32_t isz = random_size();
258
259 LCUT_TRUE(tc, ascii85_get_max_encoded_length(isz) <= (int32_t )sizeof(obuf));
260
261 for (uint32_t i = 0u; i < isz; )
262 {
263 if ((i % 8u) == 0u)
264 {
265 rand = xorshift128plus_next();
266 }
267 else
268 {
269 rand >>= 8u;
270 }
271 ibuf[i++] = rand & 0xffu;
272 }
273
274 int32_t olen = encode_ascii85(ibuf, isz, obuf, sizeof(obuf));
275
276 LCUT_TRUE(tc, olen >= 0);
277 LCUT_TRUE(tc, olen <= ascii85_get_max_encoded_length(isz));
278
279 int32_t max_decoded = ascii85_get_max_decoded_length(olen);
280 LCUT_TRUE(tc, max_decoded > 0);
281 LCUT_TRUE(tc, max_decoded <= (int32_t )sizeof(dbuf));
282
283 olen = decode_ascii85(obuf, olen, dbuf, sizeof(dbuf));
284
285 if (olen < 0) printf("Err: %d isz: %d seed: %" PRIu64 "\n", olen, isz, rand_seed); else {}
286
287 LCUT_TRUE(tc, olen >= 0);
288 LCUT_TRUE(tc, olen <= max_decoded);
289 LCUT_TRUE(tc, (uint32_t )olen == isz);
290
291 LCUT_TRUE(tc, (0 == memcmp(dbuf, ibuf, olen)));
292 }
293}
