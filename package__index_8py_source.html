79def parse_bdist_wininst(name):
80 """Return (base,pyversion) or (None,None) for possible .exe name"""
81
82 lower = name.lower()
83 base, py_ver, plat = None, None, None
84
85 if lower.endswith('.exe'):
86 if lower.endswith('.win32.exe'):
87 base = name[:-10]
88 plat = 'win32'
89 elif lower.startswith('.win32-py', -16):
90 py_ver = name[-7:-4]
91 base = name[:-16]
92 plat = 'win32'
93 elif lower.endswith('.win-amd64.exe'):
94 base = name[:-14]
95 plat = 'win-amd64'
96 elif lower.startswith('.win-amd64-py', -20):
97 py_ver = name[-7:-4]
98 base = name[:-20]
99 plat = 'win-amd64'
100 return base, py_ver, plat

114def distros_for_url(url, metadata=None):
115 """Yield egg or source distribution objects that might be found at a URL"""
116 base, fragment = egg_info_for_url(url)
117 for dist in distros_for_location(url, base, metadata):
118 yield dist
119 if fragment:
120 match = EGG_FRAGMENT.match(fragment)
121 if match:
122 for dist in interpret_distro_name(

124 ):
125 yield dist

128def distros_for_location(location, basename, metadata=None):
129 """Yield egg or source distribution objects based on basename"""
130 if basename.endswith('.egg.zip'):
131 basename = basename[:-4] # strip the .zip
132 if basename.endswith('.egg') and '-' in basename:
133 # only one, unambiguous interpretation
134 return [Distribution.from_location(location, basename, metadata)]
135 if basename.endswith('.whl') and '-' in basename:
136 wheel = Wheel(basename)
137 if not wheel.is_compatible():
138 return []
139 return [
140 Distribution(
141 location=location,
142 project_name=wheel.project_name,
143 version=wheel.version,
144 # Increase priority over eggs.
145 precedence=EGG_DIST + 1,
146 )
147 ]
148 if basename.endswith('.exe'):
149 win_base, py_ver, platform = parse_bdist_wininst(basename)
150 if win_base is not None:
151 return interpret_distro_name(
152 location, win_base, metadata, py_ver, BINARY_DIST, platform
153 )
154 # Try source distro extensions (.zip, .tgz, etc.)
155 #
156 for ext in EXTENSIONS:
157 if basename.endswith(ext):
158 basename = basename[: -len(ext)]
159 return interpret_distro_name(location, basename, metadata)
160 return [] # no extension matched

170def interpret_distro_name(
171 location, basename, metadata, py_version=None, precedence=SOURCE_DIST, platform=None
172):
173 """Generate the interpretation of a source distro name
174
175 Note: if `location` is a filesystem filename, you should call
176 ``pkg_resources.normalize_path()`` on it before passing it to this
177 routine!
178 """
179
180 parts = basename.split('-')
181 if not py_version and any(re.match(r'py\d\.\d$', p) for p in parts[2:]):
182 # it is a bdist_dumb, not an sdist -- bail out
183 return
184
185 # find the pivot (p) that splits the name from the version.
186 # infer the version as the first item that has a digit.
187 for p in range(len(parts)):
188 if parts[p][:1].isdigit():
189 break
190 else:
191 p = len(parts)
192
193 yield Distribution(
194 location,
195 metadata,
196 '-'.join(parts[:p]),
197 '-'.join(parts[p:]),
198 py_version=py_version,
199 precedence=precedence,
200 platform=platform
201 )

224def find_external_links(url, page):
225 """Find rel="homepage" and rel="download" links in `page`, yielding URLs"""
226
227 for match in REL.finditer(page):
228 tag, rel = match.groups()
229 rels = set(map(str.strip, rel.lower().split(',')))

267class HashChecker(ContentChecker):
268 pattern = re.compile(
269 r'(?P<hash_name>sha1|sha224|sha384|sha256|sha512|md5)='
270 r'(?P<expected>[a-f0-9]+)'

273 def __init__(self, hash_name, expected):
274 self.hash_name = hash_name
275 self.hash = hashlib.new(hash_name)
276 self.expected = expected

278 @classmethod
279 def from_url(cls, url):
280 "Construct a (possibly null) ContentChecker from a URL"
281 fragment = urllib.parse.urlparse(url)[-1]
282 if not fragment:
283 return ContentChecker()
284 match = cls.pattern.search(fragment)
285 if not match:
286 return ContentChecker()
287 return cls(**match.groupdict())

330 def process_url(self, url, retrieve=False): # noqa: C901
331 """Evaluate a URL as a possible download, and maybe retrieve it"""
332 if url in self.scanned_urls and not retrieve:
333 return
334 self.scanned_urls[url] = True
335 if not URL_SCHEME(url):
336 self.process_filename(url)
337 return
338 else:
339 dists = list(distros_for_url(url))
340 if dists:
341 if not self.url_ok(url):
342 return
343 self.debug("Found link: %s", url)
344
345 if dists or not retrieve or url in self.fetched_urls:
346 list(map(self.addadd, dists))
347 return # don't need the actual page
348
349 if not self.url_ok(url):
350 self.fetched_urls[url] = True
351 return
352
353 self.info("Reading %s", url)
354 self.fetched_urls[url] = True # prevent multiple fetch attempts
355 tmpl = "Download error on %s: %%s -- Some packages may not be found!"
356 f = self.open_url(url, tmpl % url)
357 if f is None:
358 return
359 if isinstance(f, urllib.error.HTTPError) and f.code == 401:
360 self.info("Authentication error: %s" % f.msg)
361 self.fetched_urls[f.url] = True
362 if 'html' not in f.headers.get('content-type', '').lower():
363 f.close() # not html, we can't process it
364 return
365
366 base = f.url # handle redirects
367 page = f.read()
368 if not isinstance(page, str):
369 # In Python 3 and got bytes but want str.
370 if isinstance(f, urllib.error.HTTPError):
371 # Errors have no charset, assume latin1:
372 charset = 'latin-1'
373 else:
374 charset = f.headers.get_param('charset') or 'latin-1'
375 page = page.decode(charset, "ignore")
376 f.close()
377 for match in HREF.finditer(page):
378 link = urllib.parse.urljoin(base, htmldecode(match.group(1)))
379 self.process_url(link)
380 if url.startswith(self.index_url) and getattr(f, 'code', None) != 404:
381 page = self.process_index(url, page)

439 def _scan(self, link):
440 # Process a URL to see if it's for a package page
441 NO_MATCH_SENTINEL = None, None
442 if not link.startswith(self.index_url):
443 return NO_MATCH_SENTINEL
444
445 parts = list(map(urllib.parse.unquote, link[len(self.index_url) :].split('/')))
446 if len(parts) != 2 or '#' in parts[1]:
447 return NO_MATCH_SENTINEL
448
449 # it's a package page, sanitize and index it
450 pkg = safe_name(parts[0])
451 ver = safe_version(parts[1])
452 self.package_pages.setdefault(pkg.lower(), {})[link] = True
453 return to_filename(pkg), to_filename(ver)

455 def process_index(self, url, page):
456 """Process the contents of a PyPI page"""
457
458 # process an index page into the package-page index
459 for match in HREF.finditer(page):
460 try:
461 self._scan(urllib.parse.urljoin(url, htmldecode(match.group(1))))
462 except ValueError:
463 pass
464
465 pkg, ver = self._scan(url) # ensure this page is in the page index
466 if not pkg:
467 return "" # no sense double-scanning non-package pages
468
469 # process individual package page
470 for new_url in find_external_links(url, page):
471 # Process the found URL
472 base, frag = egg_info_for_url(new_url)
473 if base.endswith('.py') and not frag:
474 if ver:
475 new_url += '#egg=%s-%s' % (pkg, ver)
476 else:
477 self.need_version_info(url)
478 self.scan_url(new_url)
479
480 return PYPI_MD5.sub(
481 lambda m: '<a href="%s#md5=%s">%s</a>' % m.group(1, 3, 2), page
482 )

568 def download(self, spec, tmpdir):
569 """Locate and/or download `spec` to `tmpdir`, returning a local path
570
571 `spec` may be a ``Requirement`` object, or a string containing a URL,
572 an existing local filename, or a project/version requirement spec
573 (i.e. the string form of a ``Requirement`` object). If it is the URL
574 of a .py file with an unambiguous ``#egg=name-version`` tag (i.e., one
575 that escapes ``-`` as ``_`` throughout), a trivial ``setup.py`` is
576 automatically created alongside the downloaded file.
577
578 If `spec` is a ``Requirement`` object or a string containing a
579 project/version requirement spec, this method returns the location of
580 a matching distribution (possibly after downloading it to `tmpdir`).
581 If `spec` is a locally existing file or directory name, it is simply
582 returned unchanged. If `spec` is a URL, it is downloaded to a subpath
583 of `tmpdir`, and the local filename is returned. Various errors may be
584 raised if a problem occurs during downloading.
585 """
586 if not isinstance(spec, Requirement):
587 scheme = URL_SCHEME(spec)
588 if scheme:
589 # It's a url, download it to tmpdir
590 found = self._download_url(spec, tmpdir)
591 base, fragment = egg_info_for_url(spec)
592 if base.endswith('.py'):
593 found = self.gen_setup(found, fragment, tmpdir)
594 return found
595 elif os.path.exists(spec):
596 # Existing file or directory, just return it
597 return spec
598 else:
599 spec = parse_requirement_arg(spec)
600 return getattr(self.fetch_distribution(spec, tmpdir), 'location', None)

602 def fetch_distribution( # noqa: C901 # is too complex (14) # FIXME
603 self,
604 requirement,
605 tmpdir,
606 force_scan=False,
607 source=False,
608 develop_ok=False,
609 local_index=None,
610 ):
611 """Obtain a distribution suitable for fulfilling `requirement`
612
613 `requirement` must be a ``pkg_resources.Requirement`` instance.
614 If necessary, or if the `force_scan` flag is set, the requirement is
615 searched for in the (online) package index as well as the locally
616 installed packages. If a distribution matching `requirement` is found,
617 the returned distribution's ``location`` is the value you would have
618 gotten from calling the ``download()`` method with the matching
619 distribution's URL or filename. If no matching distribution is found,
620 ``None`` is returned.
621
622 If the `source` flag is set, only source distributions and source
623 checkout links will be considered. Unless the `develop_ok` flag is
624 set, development and system eggs (i.e., those using the ``.egg-info``
625 format) will be ignored.
626 """
627 # process a Requirement
628 self.info("Searching for %s", requirement)
629 skipped = {}
630 dist = None
631
632 def find(req, env=None):
633 if env is None:
634 env = self
635 # Find a matching distribution; may be called more than once
636
637 for dist in env[req.key]:
638
639 if dist.precedence == DEVELOP_DIST and not develop_ok:
640 if dist not in skipped:
641 self.warn(
642 "Skipping development or system egg: %s",
643 dist,
644 )
645 skipped[dist] = 1
646 continue
647
648 test = dist in req and (dist.precedence <= SOURCE_DIST or not source)
649 if test:
650 loc = self.download(dist.location, tmpdir)
651 dist.download_location = loc
652 if os.path.exists(dist.download_location):
653 return dist
654
655 if force_scan:
656 self.prescan()
657 self.find_packages(requirement)
658 dist = find(requirement)
659
660 if not dist and local_index is not None:
661 dist = find(requirement, local_index)
662
663 if dist is None:
664 if self.to_scan is not None:
665 self.prescan()
666 dist = find(requirement)
667
668 if dist is None and not force_scan:
669 self.find_packages(requirement)
670 dist = find(requirement)
671
672 if dist is None:
673 self.warn(
674 "No local packages or working download links found for %s%s",
675 (source and "a source distribution of " or ""),
676 requirement,
677 )
678 else:
679 self.info("Best match: %s", dist)
680 return dist.clone(location=dist.download_location)

695 def gen_setup(self, filename, fragment, tmpdir):
696 match = EGG_FRAGMENT.match(fragment)
697 dists = (
698 match
699 and [
700 d
701 for d in interpret_distro_name(filename, match.group(1), None)
702 if d.version
703 ]
704 or []
705 )
706
707 if len(dists) == 1: # unambiguous ``#egg`` fragment
708 basename = os.path.basename(filename)
709
710 # Make sure the file has been downloaded to the temp dir.
711 if os.path.dirname(filename) != tmpdir:
712 dst = os.path.join(tmpdir, basename)
713 if not (os.path.exists(dst) and os.path.samefile(filename, dst)):
714 shutil.copy2(filename, dst)
715 filename = dst
716
717 with open(os.path.join(tmpdir, 'setup.py'), 'w') as file:
718 file.write(
719 "from setuptools import setup\n"
720 "setup(name=%r, version=%r, py_modules=[%r])\n"
721 % (
722 dists[0].project_name,
723 dists[0].version,
724 os.path.splitext(basename)[0],
725 )
726 )
727 return filename
728
729 elif match:

1010def _encode_auth(auth):
1011 """
1012 Encode auth from a URL suitable for an HTTP header.
1013 >>> str(_encode_auth('username%3Apassword'))
1014 'dXNlcm5hbWU6cGFzc3dvcmQ='
1015
1016 Long auth strings should not cause a newline to be inserted.
1017 >>> long_auth = 'username:' + 'password'*10
1018 >>> chr(10) in str(_encode_auth(long_auth))
1019 False
1020 """
1021 auth_s = urllib.parse.unquote(auth)
1022 # convert to bytes
1023 auth_bytes = auth_s.encode()
1024 encoded_bytes = base64.b64encode(auth_bytes)
1025 # convert back to a string
1026 encoded = encoded_bytes.decode()
1027 # strip the trailing carriage return
1028 return encoded.replace('\n', '')
