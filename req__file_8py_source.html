126def parse_requirements(
127 filename: str,
128 session: PipSession,
129 finder: Optional["PackageFinder"] = None,
130 options: Optional[optparse.Values] = None,
131 constraint: bool = False,
132) -> Generator[ParsedRequirement, None, None]:
133 """Parse a requirements file and yield ParsedRequirement instances.
134
135 :param filename: Path or url of requirements file.
136 :param session: PipSession instance.
137 :param finder: Instance of pip.index.PackageFinder.
138 :param options: cli options.
139 :param constraint: If true, parsing a constraint file rather than
140 requirements file.
141 """
142 line_parser = get_line_parser(finder)
143 parser = RequirementsFileParser(session, line_parser)
144
145 for parsed_line in parser.parse(filename, constraint):
146 parsed_req = handle_line(
147 parsed_line, options=options, finder=finder, session=session
148 )
149 if parsed_req is not None:
150 yield parsed_req

165def handle_requirement_line(
166 line: ParsedLine,
167 options: Optional[optparse.Values] = None,
168) -> ParsedRequirement:
169
170 # preserve for the nested code path
171 line_comes_from = "{} {} (line {})".format(
172 "-c" if line.constraint else "-r",
173 line.filename,
174 line.lineno,
175 )
176
177 assert line.is_requirement
178
179 if line.is_editable:
180 # For editable requirements, we don't support per-requirement
181 # options, so just return the parsed requirement.
182 return ParsedRequirement(
183 requirement=line.requirement,
184 is_editable=line.is_editable,
185 comes_from=line_comes_from,
186 constraint=line.constraint,
187 )
188 else:
189 # get the options that apply to requirements
190 req_options = {}
191 for dest in SUPPORTED_OPTIONS_REQ_DEST:
192 if dest in line.opts.__dict__ and line.opts.__dict__[dest]:
193 req_options[dest] = line.opts.__dict__[dest]
194
195 line_source = f"line {line.lineno} of {line.filename}"
196 return ParsedRequirement(
197 requirement=line.requirement,
198 is_editable=line.is_editable,
199 comes_from=line_comes_from,
200 constraint=line.constraint,
201 options=req_options,
202 line_source=line_source,
203 )

206def handle_option_line(
207 opts: Values,
208 filename: str,
209 lineno: int,
210 finder: Optional["PackageFinder"] = None,
211 options: Optional[optparse.Values] = None,
212 session: Optional[PipSession] = None,

214

216 # percolate options upward
217 if opts.require_hashes:
218 options.require_hashes = opts.require_hashes
219 if opts.features_enabled:
220 options.features_enabled.extend(
221 f for f in opts.features_enabled if f not in options.features_enabled
222 )
223
224 # set finder options

238 # of the find_links: support a find-links local path
239 # relative to a requirements file.
240 value = opts.find_links[0]
241 req_dir = os.path.dirname(os.path.abspath(filename))
242 relative_to_reqs_file = os.path.join(req_dir, value)

270def handle_line(
271 line: ParsedLine,
272 options: Optional[optparse.Values] = None,
273 finder: Optional["PackageFinder"] = None,
274 session: Optional[PipSession] = None,
275) -> Optional[ParsedRequirement]:
276 """Handle a single parsed requirements line; This can result in
277 creating/yielding requirements, or updating the finder.
278
279 :param line: The parsed line to be processed.
280 :param options: CLI options.
281 :param finder: The finder - updated by non-requirement lines.
282 :param session: The session - updated by non-requirement lines.
283
284 Returns a ParsedRequirement object if the line is a requirement line,
285 otherwise returns None.
286
287 For lines that contain requirements, the only options that have an effect
288 are from SUPPORTED_OPTIONS_REQ, and they are scoped to the
289 requirement. Other options from SUPPORTED_OPTIONS may be present, but are
290 ignored.
291
292 For lines that do not contain requirements, the only options that have an
293 effect are from SUPPORTED_OPTIONS. Options from SUPPORTED_OPTIONS_REQ may
294 be present, but are ignored. These lines may contain multiple options
295 (although our docs imply only one is supported), and all our parsed and
296 affect the finder.
297 """
298
299 if line.is_requirement:
300 parsed_req = handle_requirement_line(line, options)
301 return parsed_req
302 else:
303 handle_option_line(
304 line.opts,
305 line.filename,
306 line.lineno,
307 finder,
308 options,
309 session,
310 )
311 return None

329 def _parse_and_recurse(
330 self, filename: str, constraint: bool
331 ) -> Generator[ParsedLine, None, None]:
332 for line in self._parse_file(filename, constraint):
333 if not line.is_requirement and (
334 line.opts.requirements or line.opts.constraints
335 ):
336 # parse a nested requirements file
337 if line.opts.requirements:
338 req_path = line.opts.requirements[0]
339 nested_constraint = False
340 else:
341 req_path = line.opts.constraints[0]
342 nested_constraint = True
343
344 # original file is over http
345 if SCHEME_RE.search(filename):
346 # do a url join so relative paths work
347 req_path = urllib.parse.urljoin(filename, req_path)
348 # original file and nested file are paths
349 elif not SCHEME_RE.search(req_path):
350 # do a join so relative paths work
351 req_path = os.path.join(
352 os.path.dirname(filename),
353 req_path,
354 )
355
356 yield from self._parse_and_recurse(req_path, nested_constraint)
357 else:
358 yield line

360 def _parse_file(
361 self, filename: str, constraint: bool
362 ) -> Generator[ParsedLine, None, None]:
363 _, content = get_file_content(filename, self._session)
364
365 lines_enum = preprocess(content)
366
367 for line_number, line in lines_enum:
368 try:
369 args_str, opts = self._line_parser(line)
370 except OptionParsingError as e:
371 # add offending line
372 msg = f"Invalid requirement: {line}\n{e.msg}"
373 raise RequirementsFileParseError(msg)
374
375 yield ParsedLine(
376 filename,
377 line_number,
378 args_str,
379 opts,
380 constraint,
381 )

384def get_line_parser(finder: Optional["PackageFinder"]) -> LineParser:
385 def parse_line(line: str) -> Tuple[str, Values]:
386 # Build new parser for each line since it accumulates appendable
387 # options.
388 parser = build_parser()
389 defaults = parser.get_default_values()
390 defaults.index_url = None
391 if finder:
392 defaults.format_control = finder.format_control
393
394 args_str, options_str = break_args_options(line)
395
396 try:
397 options = shlex.split(options_str)
398 except ValueError as e:
399 raise OptionParsingError(f"Could not split options: {options_str}") from e
400
401 opts, _ = parser.parse_args(options, defaults)
402
403 return args_str, opts
404
405 return parse_line

408def break_args_options(line: str) -> Tuple[str, str]:
409 """Break up the line into an args and options string. We only want to shlex
410 (and then optparse) the options, not the args. args can contain markers
411 which are corrupted by shlex.
412 """
413 tokens = line.split(" ")
414 args = []
415 options = tokens[:]
416 for token in tokens:
417 if token.startswith("-") or token.startswith("--"):
418 break
419 else:
420 args.append(token)
421 options.pop(0)
422 return " ".join(args), " ".join(options)

430def build_parser() -> optparse.OptionParser:
431 """
432 Return a parser for parsing requirement lines
433 """
434 parser = optparse.OptionParser(add_help_option=False)
435
436 option_factories = SUPPORTED_OPTIONS + SUPPORTED_OPTIONS_REQ
437 for option_factory in option_factories:
438 option = option_factory()
439 parser.add_option(option)
440
441 # By default optparse sys.exits on parsing errors. We want to wrap
442 # that in our own exception.
443 def parser_exit(self: Any, msg: str) -> "NoReturn":
444 raise OptionParsingError(msg)
445
446 # NOTE: mypy disallows assigning to a method
447 # https://github.com/python/mypy/issues/2427
448 parser.exit = parser_exit # type: ignore
449
450 return parser

453def join_lines(lines_enum: ReqFileLines) -> ReqFileLines:
454 """Joins a line ending in '\' with the previous line (except when following
455 comments). The joined line takes on the index of the first line.
456 """
457 primary_line_number = None
458 new_line: List[str] = []
459 for line_number, line in lines_enum:
460 if not line.endswith("\\") or COMMENT_RE.match(line):
461 if COMMENT_RE.match(line):
462 # this ensures comments are always matched later
463 line = " " + line
464 if new_line:
465 new_line.append(line)
466 assert primary_line_number is not None
467 yield primary_line_number, "".join(new_line)
468 new_line = []
469 else:
470 yield line_number, line
471 else:
472 if not new_line:
473 primary_line_number = line_number
474 new_line.append(line.strip("\\"))
475
476 # last line contains \
477 if new_line:
478 assert primary_line_number is not None
479 yield primary_line_number, "".join(new_line)
480
481 # TODO: handle space after '\'.

495def expand_env_variables(lines_enum: ReqFileLines) -> ReqFileLines:
496 """Replace all environment variables that can be retrieved via `os.getenv`.
497
498 The only allowed format for environment variables defined in the
499 requirement file is `${MY_VARIABLE_1}` to ensure two things:
500
501 1. Strings that contain a `$` aren't accidentally (partially) expanded.
502 2. Ensure consistency across platforms for requirement files.
503
504 These points are the result of a discussion on the `github pull
505 request #3514 <https://github.com/pypa/pip/pull/3514>`_.
506
507 Valid characters in variable names follow the `POSIX standard
508 <http://pubs.opengroup.org/onlinepubs/9699919799/>`_ and are limited
509 to uppercase letter, digits and the `_` (underscore).
510 """
511 for line_number, line in lines_enum:
512 for env_var, var_name in ENV_VAR_RE.findall(line):
513 value = os.getenv(var_name)
514 if not value:
515 continue
516
517 line = line.replace(env_var, value)
518
519 yield line_number, line

522def get_file_content(url: str, session: PipSession) -> Tuple[str, str]:
523 """Gets the content of a file; it may be a filename, file: URL, or
524 http: URL. Returns (location, content). Content is unicode.
525 Respects # -*- coding: declarations on the retrieved files.
526
527 :param url: File path or url.
528 :param session: PipSession instance.
529 """
530 scheme = get_url_scheme(url)
531
532 # Pip has special support for file:// URLs (LocalFSAdapter).
533 if scheme in ["http", "https", "file"]:
534 resp = session.get(url)
535 raise_for_status(resp)
536 return resp.url, resp.text
537
538 # Assume this is a bare path.
539 try:
540 with open(url, "rb") as f:
541 content = auto_decode(f.read())
542 except OSError as exc:
543 raise InstallationError(f"Could not open requirements file: {exc}")
544 return url, content
