34class wait_base(abc.ABC):
35 """Abstract base class for wait strategies."""
36
37 @abc.abstractmethod
38 def __call__(self, retry_state: "RetryCallState") -> float:
39 pass
40
41 def __add__(self, other: "wait_base") -> "wait_combine":
42 return wait_combine(self, other)
43
44 def __radd__(self, other: "wait_base") -> typing.Union["wait_combine", "wait_base"]:
45 # make it possible to use multiple waits with the built-in sum function
46 if other == 0:
47 return self
48 return self.__add__(other)

68class wait_random(wait_base):
69 """Wait strategy that waits a random amount of time between min/max."""
70
71 def __init__(self, min: wait_unit_type = 0, max: wait_unit_type = 1) -> None: # noqa
72 self.wait_random_min = to_seconds(min)
73 self.wait_random_max = to_seconds(max)

75 def __call__(self, retry_state: "RetryCallState") -> float:
76 return self.wait_random_min + (random.random() * (self.wait_random_max - self.wait_random_min))

89class wait_chain(wait_base):
90 """Chain two or more waiting strategies.
91
92 If all strategies are exhausted, the very last strategy is used
93 thereafter.
94
95 For example::
96
97 @retry(wait=wait_chain(*[wait_fixed(1) for i in range(3)] +
98 [wait_fixed(2) for j in range(5)] +
99 [wait_fixed(5) for k in range(4)))
100 def wait_chained():
101 print("Wait 1s for 3 attempts, 2s for 5 attempts and 5s
102 thereafter.")
103 """
104
105 def __init__(self, *strategies: wait_base) -> None:
106 self.strategies = strategies
107
108 def __call__(self, retry_state: "RetryCallState") -> float:
109 wait_func_no = min(max(retry_state.attempt_number, 1), len(self.strategies))
110 wait_func = self.strategies[wait_func_no - 1]
111 return wait_func(retry_state=retry_state)

114class wait_incrementing(wait_base):
115 """Wait an incremental amount of time after each attempt.
116
117 Starting at a starting value and incrementing by a value for each attempt
118 (and restricting the upper limit to some maximum value).
119 """
120
121 def __init__(
122 self,
123 start: wait_unit_type = 0,
124 increment: wait_unit_type = 100,
125 max: wait_unit_type = _utils.MAX_WAIT, # noqa
126 ) -> None:
127 self.start = to_seconds(start)
128 self.increment = to_seconds(increment)
129 self.max = to_seconds(max)
130
131 def __call__(self, retry_state: "RetryCallState") -> float:
132 result = self.start + (self.increment * (retry_state.attempt_number - 1))
133 return max(0, min(result, self.max))

136class wait_exponential(wait_base):
137 """Wait strategy that applies exponential backoff.
138
139 It allows for a customized multiplier and an ability to restrict the
140 upper and lower limits to some maximum and minimum value.
141
142 The intervals are fixed (i.e. there is no jitter), so this strategy is
143 suitable for balancing retries against latency when a required resource is
144 unavailable for an unknown duration, but *not* suitable for resolving
145 contention between multiple processes for a shared resource. Use
146 wait_random_exponential for the latter case.
147 """
148
149 def __init__(
150 self,
151 multiplier: typing.Union[int, float] = 1,
152 max: wait_unit_type = _utils.MAX_WAIT, # noqa
153 exp_base: typing.Union[int, float] = 2,
154 min: wait_unit_type = 0, # noqa
155 ) -> None:
156 self.multiplier = multiplier
157 self.min = to_seconds(min)
158 self.max = to_seconds(max)
159 self.exp_base = exp_base
160
161 def __call__(self, retry_state: "RetryCallState") -> float:
162 try:
163 exp = self.exp_base ** (retry_state.attempt_number - 1)
164 result = self.multiplier * exp
165 except OverflowError:
166 return self.max
167 return max(max(0, self.min), min(result, self.max))

170class wait_random_exponential(wait_exponential):
171 """Random wait with exponentially widening window.
172
173 An exponential backoff strategy used to mediate contention between multiple
174 uncoordinated processes for a shared resource in distributed systems. This
175 is the sense in which "exponential backoff" is meant in e.g. Ethernet
176 networking, and corresponds to the "Full Jitter" algorithm described in
177 this blog post:
178
179 https://aws.amazon.com/blogs/architecture/exponential-backoff-and-jitter/
180
181 Each retry occurs at a random time in a geometrically expanding interval.
182 It allows for a custom multiplier and an ability to restrict the upper
183 limit of the random interval to some maximum value.
184
185 Example::
186
187 wait_random_exponential(multiplier=0.5, # initial window 0.5s
188 max=60) # max 60s timeout
189
190 When waiting for an unavailable resource to become available again, as
191 opposed to trying to resolve contention for a shared resource, the
192 wait_exponential strategy (which uses a fixed interval) may be preferable.
193
194 """
195
196 def __call__(self, retry_state: "RetryCallState") -> float:
197 high = super().__call__(retry_state=retry_state)
198 return random.uniform(0, high)

201class wait_exponential_jitter(wait_base):
202 """Wait strategy that applies exponential backoff and jitter.
203
204 It allows for a customized initial wait, maximum wait and jitter.
205
206 This implements the strategy described here:
207 https://cloud.google.com/storage/docs/retry-strategy
208
209 The wait time is min(initial * (2**n + random.uniform(0, jitter)), maximum)
210 where n is the retry count.
211 """
212
213 def __init__(
214 self,
215 initial: float = 1,
216 max: float = _utils.MAX_WAIT, # noqa
217 exp_base: float = 2,
218 jitter: float = 1,
219 ) -> None:
220 self.initial = initial
221 self.max = max
222 self.exp_base = exp_base
223 self.jitter = jitter
224
225 def __call__(self, retry_state: "RetryCallState") -> float:
226 jitter = random.uniform(0, self.jitter)
227 try:
228 exp = self.exp_base ** (retry_state.attempt_number - 1)
229 result = self.initial * exp + jitter
230 except OverflowError:
231 result = self.max
232 return max(0, min(result, self.max))
