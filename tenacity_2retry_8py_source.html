59class retry_if_exception(retry_base):

120class retry_if_exception_cause_type(retry_base):
121 """Retries if any of the causes of the raised exception is of one or more types.
122
123 The check on the type of the cause of the exception is done recursively (until finding
124 an exception in the chain that has no `__cause__`)
125 """
126
127 def __init__(
128 self,
129 exception_types: typing.Union[
130 typing.Type[BaseException],
131 typing.Tuple[typing.Type[BaseException], ...],
132 ] = Exception,
133 ) -> None:
134 self.exception_cause_types = exception_types
135
136 def __call__(self, retry_state: "RetryCallState") -> bool:
137 if retry_state.outcome.failed:
138 exc = retry_state.outcome.exception()
139 while exc is not None:
140 if isinstance(exc.__cause__, self.exception_cause_types):
141 return True
142 exc = exc.__cause__
143
144 return False

173class retry_if_exception_message(retry_if_exception):
174 """Retries if an exception message equals or matches."""
175
176 def __init__(
177 self,
178 message: typing.Optional[str] = None,
179 match: typing.Optional[str] = None,
180 ) -> None:
181 if message and match:
182 raise TypeError(f"{self.__class__.__name__}() takes either 'message' or 'match', not both")
183
184 # set predicate
185 if message:
186
187 def message_fnc(exception: BaseException) -> bool:
188 return message == str(exception)
189
190 predicate = message_fnc
191 elif match:
192 prog = re.compile(match)
193
194 def match_fnc(exception: BaseException) -> bool:
195 return bool(prog.match(str(exception)))
196
197 predicate = match_fnc
198 else:
199 raise TypeError(f"{self.__class__.__name__}() missing 1 required argument 'message' or 'match'")
200
201 super().__init__(predicate)
