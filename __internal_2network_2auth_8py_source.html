69 def get_auth_info(self, url: str, username: Optional[str]) -> Optional[AuthInfo]:
70 # Support keyring's get_credential interface which supports getting
71 # credentials without a username. This is only available for
72 # keyring>=15.2.0.

74 logger.debug("Getting credentials from keyring for %s", url)

103 def get_auth_info(self, url: str, username: Optional[str]) -> Optional[AuthInfo]:
104 # This is the default implementation of keyring.get_credential
105 # https://github.com/jaraco/keyring/blob/97689324abcf01bd1793d49063e7ca01e03d7d07/keyring/backend.py#L134-L139

107 password = self._get_password(url, username)

115 def _get_password(self, service_name: str, username: str) -> Optional[str]:
116 """Mirror the implementation of keyring.get_password using cli"""
117 if self.keyring is None:
118 return None
119
120 cmd = [self.keyring, "get", service_name, username]
121 env = os.environ.copy()
122 env["PYTHONIOENCODING"] = "utf-8"
123 res = subprocess.run(
124 cmd,
125 stdin=subprocess.DEVNULL,
126 capture_output=True,
127 env=env,
128 )
129 if res.returncode:
130 return None
131 return res.stdout.decode("utf-8").strip(os.linesep)
132
133 def _set_password(self, service_name: str, username: str, password: str) -> None:
134 """Mirror the implementation of keyring.set_password using cli"""
135 if self.keyring is None:
136 return None
137
138 cmd = [self.keyring, "set", service_name, username]
139 input_ = (password + os.linesep).encode("utf-8")
140 env = os.environ.copy()
141 env["PYTHONIOENCODING"] = "utf-8"
142 res = subprocess.run(cmd, input=input_, env=env)
143 res.check_returncode()
144 return None

147def get_keyring_provider() -> KeyRingBaseProvider:
148 # keyring has previously failed and been disabled
149 if not KEYRING_DISABLED:
150 # Default to trying to use Python provider
151 try:
152 return KeyRingPythonProvider()
153 except ImportError:
154 pass
155 except Exception as exc:
156 # In the event of an unexpected exception
157 # we should warn the user

160 "trying to find a keyring executable as a fallback",
161 str(exc),
162 )
163
164 # Fallback to Cli Provider if `keyring` isn't installed
165 cli = shutil.which("keyring")

191class MultiDomainBasicAuth(AuthBase):
192 def __init__(
193 self, prompting: bool = True, index_urls: Optional[List[str]] = None
194 ) -> None:
195 self.prompting = prompting
196 self.index_urls = index_urls
197 self.passwords: Dict[str, AuthInfo] = {}
198 # When the user is prompted to enter credentials and keyring is
199 # available, we will offer to save them. If the user accepts,
200 # this value is set to the credentials they entered. After the
201 # request authenticates, the caller should call
202 # ``save_credentials`` to save these.
203 self._credentials_to_save: Optional[Credentials] = None

205 def _get_index_url(self, url: str) -> Optional[str]:
206 """Return the original index URL matching the requested URL.
207
208 Cached or dynamically generated credentials may work against
209 the original index URL rather than just the netloc.
210
211 The provided url should have had its username and password
212 removed already. If the original index url had credentials then
213 they will be included in the return value.
214
215 Returns None if no matching index was found, or if --no-index
216 was specified by the user.
217 """
218 if not url or not self.index_urls:
219 return None
220
221 for u in self.index_urls:
222 prefix = remove_auth_from_url(u).rstrip("/") + "/"
223 if url.startswith(prefix):
224 return u
225 return None

227 def _get_new_credentials(
228 self,
229 original_url: str,
230 allow_netrc: bool = True,
231 allow_keyring: bool = False,
232 ) -> AuthInfo:
233 """Find and return credentials for the specified URL."""
234 # Split the credentials and netloc from the url.
235 url, netloc, url_user_password = split_auth_netloc_from_url(
236 original_url,
237 )
238
239 # Start with the credentials embedded in the url
240 username, password = url_user_password
241 if username is not None and password is not None:
242 logger.debug("Found credentials in url for %s", netloc)
243 return url_user_password
244
245 # Find a matching index url for this request
246 index_url = self._get_index_url(url)
247 if index_url:
248 # Split the credentials from the url.
249 index_info = split_auth_netloc_from_url(index_url)
250 if index_info:
251 index_url, _, index_url_user_password = index_info
252 logger.debug("Found index url %s", index_url)
253
254 # If an index URL was found, try its embedded credentials
255 if index_url and index_url_user_password[0] is not None:
256 username, password = index_url_user_password
257 if username is not None and password is not None:
258 logger.debug("Found credentials in index url for %s", netloc)
259 return index_url_user_password
260
261 # Get creds from netrc if we still don't have them
262 if allow_netrc:
263 netrc_auth = get_netrc_auth(original_url)
264 if netrc_auth:
265 logger.debug("Found credentials in netrc for %s", netloc)
266 return netrc_auth
267
268 # If we don't have a password and keyring is available, use it.
269 if allow_keyring:
270 # The index url is more specific than the netloc, so try it first
271 # fmt: off
272 kr_auth = (
273 get_keyring_auth(index_url, username) or
274 get_keyring_auth(netloc, username)
275 )
276 # fmt: on
277 if kr_auth:
278 logger.debug("Found credentials in keyring for %s", netloc)
279 return kr_auth
280
281 return username, password

283 def _get_url_and_credentials(
284 self, original_url: str
285 ) -> Tuple[str, Optional[str], Optional[str]]:
286 """Return the credentials to use for the provided URL.
287
288 If allowed, netrc and keyring may be used to obtain the
289 correct credentials.
290
291 Returns (url_without_credentials, username, password). Note
292 that even if the original URL contains credentials, this
293 function may return a different username and password.
294 """
295 url, netloc, _ = split_auth_netloc_from_url(original_url)
296
297 # Try to get credentials from original url
298 username, password = self._get_new_credentials(original_url)
299
300 # If credentials not found, use any stored credentials for this netloc.
301 # Do this if either the username or the password is missing.
302 # This accounts for the situation in which the user has specified
303 # the username in the index url, but the password comes from keyring.
304 if (username is None or password is None) and netloc in self.passwords:
305 un, pw = self.passwords[netloc]
306 # It is possible that the cached credentials are for a different username,
307 # in which case the cache should be ignored.
308 if username is None or username == un:
309 username, password = un, pw
310
311 if username is not None or password is not None:
312 # Convert the username and password if they're None, so that
313 # this netloc will show up as "cached" in the conditional above.
314 # Further, HTTPBasicAuth doesn't accept None, so it makes sense to
315 # cache the value that is going to be used.
316 username = username or ""
317 password = password or ""
318
319 # Store any acquired credentials.
320 self.passwords[netloc] = (username, password)
321
322 assert (
323 # Credentials were found
324 (username is not None and password is not None)
325 # Credentials were not found
326 or (username is None and password is None)
327 ), f"Could not load credentials from url: {original_url}"
328
329 return url, username, password

331 def __call__(self, req: Request) -> Request:
332 # Get credentials for this request
333 url, username, password = self._get_url_and_credentials(req.url)
334
335 # Set the url of the request to the url without any credentials
336 req.url = url
337
338 if username is not None and password is not None:
339 # Send the basic auth with this request
340 req = HTTPBasicAuth(username, password)(req)
341
342 # Attach a hook to handle 401 responses
343 req.register_hook("response", self.handle_401)
344
345 return req

366 def handle_401(self, resp: Response, **kwargs: Any) -> Response:
367 # We only care about 401 responses, anything else we want to just
368 # pass through the actual response
369 if resp.status_code != 401:
370 return resp
371
372 # We are not able to prompt the user so simply return the response
373 if not self.prompting:
374 return resp
375
376 parsed = urllib.parse.urlparse(resp.url)
377
378 # Query the keyring for credentials:
379 username, password = self._get_new_credentials(
380 resp.url,
381 allow_netrc=False,
382 allow_keyring=True,
383 )
384
385 # Prompt the user for a new username and password
386 save = False
387 if not username and not password:
388 username, password, save = self._prompt_for_password(parsed.netloc)
389
390 # Store the new username and password to use for future requests
391 self._credentials_to_save = None
392 if username is not None and password is not None:
393 self.passwords[parsed.netloc] = (username, password)
394
395 # Prompt to save the password to keyring
396 if save and self._should_save_password_to_keyring():
397 self._credentials_to_save = Credentials(
398 url=parsed.netloc,
399 username=username,
400 password=password,
401 )
402
403 # Consume content and release the original connection to allow our new
404 # request to reuse the same one.
405 resp.content
406 resp.raw.release_conn()
407
408 # Add our new username and password to the request
409 req = HTTPBasicAuth(username or "", password or "")(resp.request)
410 req.register_hook("response", self.warn_on_401)
411
412 # On successful request, save the credentials that were used to
413 # keyring. (Note that if the user responded "no" above, this member
414 # is not set and nothing will be saved.)
415 if self._credentials_to_save:
416 req.register_hook("response", self.save_credentials)
417
418 # Send our new request
419 new_resp = resp.connection.send(req, **kwargs)
420 new_resp.history.append(resp)
421
422 return new_resp
