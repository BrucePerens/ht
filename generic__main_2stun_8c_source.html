9// The STUN RFC 8489 requires attributes connected with authentication, and requires
10// knowledge of the cleartext password on the server to calculate the MESSAGE-INTEGRITY
11// attribute, or at least an MD5 of username:password to calculate the
12// MESSAGE-INTEGRITY-SHA256 attribute. Public STUN servers would not have this data,
13// and I assume they do not insist on those elements. Having a cleartext password on
14// the server is not a good security practice. The MD5 derivative used in
15// MESSAGE-INTEGRITY-SHA256 is also problematical because the cleartext password is
16// necessary to produce it, and because MD5 is no longer considered a secure hash
17// algorithm.
18//
19// The RFC says nothing about how to run a public STUN server, yet many exist. Thus,
20// although required by the RFC, this client doesn't send USERNAME, USERHASH,
21// MESSAGE-INTEGRITY, MESSAGE-INTEGRITY-SHA256m because a server that doesn't have
22// any password data could not authenticate them. It sends FINGERPRINT, and it could
23// send REALM, although the REALM information would be arbitrary and probably useless.
24

154static void
155decode_mapped_address(struct stun_attribute * a, struct sockaddr * address)
156{
157 if ( a->value.mapped_address.family == 1 ) {
158 struct sockaddr_in * in = (struct sockaddr_in *)address;
159 memset(in, '\0', sizeof(*in));
160 in->sin_family = AF_INET;
161 in->sin_port = a->value.mapped_address.port;
162 in->sin_addr.s_addr = a->value.mapped_address.ipv4;
163 }
164 else {
165 struct sockaddr_in6 * in6 = (struct sockaddr_in6 *)address;
166 memset(in6, '\0', sizeof(*in6));
167 in6->sin6_family = AF_INET6;
168 in6->sin6_port = a->value.mapped_address.port;
169 memcpy(&in6->sin6_addr.s6_addr, a->value.mapped_address.ipv6.s6_addr, sizeof(in6->sin6_addr.s6_addr));
170 }

185static void
186decode_xor_mapped_address(struct stun_attribute * a, struct stun_message * message, struct sockaddr * address)
187{
188 if ( a->value.mapped_address.family == 1 ) {
189 fflush(stderr);
190 sleep(1);
191 struct sockaddr_in * in = (struct sockaddr_in *)address;
192 memset(in, '\0', sizeof(*in));
193 in->sin_family = AF_INET;
194 in->sin_port = a->value.mapped_address.port ^ ((stun_magic >> 16) & 0xffff);
195 in->sin_addr.s_addr = a->value.mapped_address.ipv4 ^ stun_magic;
196 }
197 else {
198 struct sockaddr_in6 * in6 = (struct sockaddr_in6 *)address;
199 memset(in6, '\0', sizeof(*in6));
200 in6->sin6_family = AF_INET6;
201 uint32_t * out = (uint32_t *)&in6->sin6_addr.s6_addr;
202 uint32_t * in = (uint32_t *)&a->value.mapped_address.ipv6.s6_addr;
203 for ( int i = 0; i < 4; i++ ) {
204 out[i] = in[i] ^ stun_magic;
205 }
206 }

256static int
257send_stun_request(bool ipv6)
258{
259 uint32_t send_buffer[128] = {};
260 struct stun_message * const send_packet = (struct stun_message *)send_buffer;
261 struct addrinfo * send_address;
262 ssize_t send_result;
263 unsigned int message_class = STUN_REQUEST;
264 unsigned int method = STUN_BINDING;
265 const struct stun_server * servers;
266 size_t count;
267 const struct stun_server * server;
268
269 if ( ipv6 ) {
270 servers = ipv6_servers;
271 count = ipv6_table_count;
272 }
273 else {
274 servers = ipv4_servers;
275 count = ipv4_table_count;
276 }
277
278 server = &servers[gm_choose_one(count)];
279
280 if ( (send_address = get_address(server->host, server->port, ipv6)) == 0 )
281 return -1;
282
283 if ( stun_sock >= 0 )
284 close(stun_sock);
285
286 stun_sock = socket(send_address->ai_family, SOCK_DGRAM, send_address->ai_protocol);
287 if ( stun_sock < 0 ) {
288 GM_FAIL("STUN: Can't get socket: %s\n", strerror(errno));
289 return -1;
290 }
291
292 // Bind the request address to a local address. This probably only matters
293 // for IPV6, as the only IPV4 address is usually behind NAT.
294 if ( ipv6 ) {
295 struct sockaddr_in6 * address = &GM.sta.ip6.link_local;
296 if ( gm_all_zeroes(address, sizeof(struct sockaddr_in6)) )
297 address = &GM.sta.ip6.site_local;
298 if ( gm_all_zeroes(address, sizeof(struct sockaddr_in6)) )
299 address = &GM.sta.ip6.site_unique;
300 else
301 address = &GM.sta.ip6.global[0];
302
303 (void) bind(stun_sock, (struct sockaddr *)address, sizeof(struct sockaddr_in6));
304 }
305 else
306 (void) bind(stun_sock, (struct sockaddr *)&GM.sta.ip4.address, sizeof(GM.sta.ip4.address));
307
308 send_packet->magic_cookie = stun_magic;
309 send_packet->type = htons(((message_class & 0x1) << 4) | ((message_class & 0x2) << 8) | (method & 0xf));
310 esp_fill_random(send_packet->transaction_id, sizeof(send_packet->transaction_id));
311
312 send_result = sendto(
313 stun_sock,
314 send_packet,
315 send_packet->length + 20,
316 0,
317 send_address->ai_addr,
318 send_address->ai_addrlen);
319
320 freeaddrinfo(send_address);
321
322 if ( send_result < (send_packet->length + 20) ) {
323 GM_FAIL("STUN: Send error: %d %s.\n", send_result, strerror(errno));
324 close(stun_sock);
325 stun_sock = -1;
326 return -1;
327 }
328 return stun_sock;
329}

331static int
332process_received_packet(struct stun_message * receive_packet, struct sockaddr * address, ssize_t receive_result)
333{
334 bool got_xor_mapped_address = false;
335 bool got_an_address = false;
336 struct stun_attribute * attribute = (struct stun_attribute *)receive_packet->attributes;
337 uint16_t attribute_size = ntohs(receive_packet->length);
338 if ( attribute_size < (receive_result - 20) ) {
339 GM_FAIL("STUN: packet was truncated.\n");
340 return -1;
341 }
342
343 while ( attribute_size > 0 ) {
344 uint16_t type = ntohs(attribute->type);
345 uint16_t length = ntohs(attribute->length);
346 if ( length == 0 || length >= 768 ) {
347 GM_FAIL("STUN: attribute length %d is invalid.\n", length);
348 }
349 switch ( type ) {
350 case MAPPED_ADDRESS:
351 if ( !got_xor_mapped_address ) {
352 got_an_address = true;
353 decode_mapped_address(attribute, address);
354 // char buffer[INET6_ADDRSTRLEN + 1];
355 // gm_ntop(address, buffer, sizeof(buffer));
356 }
357 break;
358 case ERROR_CODE:
359 decode_error_code(attribute);
360 return -1;
361 break;
362 case UNKNOWN_ATTRIBUTES:
363 decode_unknown_attributes(attribute);
364 break;
365 case XOR_MAPPED_ADDRESS:
366 got_an_address = true;
367 got_xor_mapped_address = true;
368 decode_xor_mapped_address(attribute, receive_packet, address);
369 break;
370 case SOFTWARE:
371 decode_software(attribute);
372 break;
373 case ALTERNATE_SERVER:
374 decode_alternate_server(attribute);
375 break;
376 case FINGERPRINT:
377 decode_fingerprint(attribute);
378 break;
379 default:
380 break;
381 }
382 uint16_t odd = length & 3;
383 unsigned int increment = length + 4;
384
385 // Pad strings to 32-bit boundaries.
386 if ( odd )
387 increment += (4 - odd);
388
389 if ( increment > attribute_size ) {
390 GM_FAIL("STUN: Attribute size mismatch.\n");
391 return -1;
392 }
393 attribute = (struct stun_attribute *)((uint8_t *)attribute + increment);
394 attribute_size -= increment;
395 }
396 if ( got_an_address )
397 return 0;
398 else {
399 GM_FAIL("STUN: message didn't include an address.\n");
400 return -1;
401 }
402}

452static void
453stun_receive(int fd, void * data, bool readable, bool writable, bool exception, bool timeout)
454{
455 struct stun_run * run = (struct stun_run *)data;
456
457 if ( readable ) {
458 int status = receive_stun_response(fd, run->address);
459
460 if ( status == 0 ) {
461 gm_fd_unregister(fd);
462 close(fd);
463 stun_sock = -1;
464 if ( run->after )
465 (run->after)(true, run->ipv6, run->address);
466 free(run);
467 return;
468 }
469 }
470 if ( timeout || exception ) {
471 gm_fd_unregister(fd);
472 close(fd);
473 stun_sock = -1;
474 }
475 if ( run->tries >= 5 ) {
476 if ( run->after )
477 (run->after)(false, run->ipv6, run->address);
478 free(run);
479 return;
480 }
481
482 run->tries++;
483
484 gm_run(stun_send, run, GM_FAST);
485}
