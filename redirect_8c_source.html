46static void
47parse_request(struct request * const r, size_t size) {
48 const char * const first_line = strstr(r->raw, crlf);
49 if ( first_line == NULL )
50 return;
51 const char * first_space = strchr(r->raw, ' ');
52 if ( first_space == NULL )
53 return;
54
55 r->path = first_space + 1;
56
57 const char * s = first_line + 2;
58 do {
59 if ( strcmp(s, "\r\n") == 0 ) {
60 r->end = s;
61 break;
62 }
63 const char * const colon_space = strstr(s, ": ");
64 if ( colon_space == NULL )
65 break;
66
67 const char * const line_end = strstr(colon_space, crlf);
68 if ( line_end == NULL )
69 break;
70
71 if ( strncasecmp(s, "Connection: keep-alive", line_end - s) == 0 ) {
72 r->keep_alive = true;
73 }
74 else if ( strncasecmp(s, "Host: ", 6) == 0 ) {
75 r->host = colon_space + 2;
76 }
77 s = line_end + 2;
78 } while ( s < r->next );
79}
