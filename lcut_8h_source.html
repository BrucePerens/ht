135/*
136 * LCUT_TEST_BEGIN and LCUT_TEST_END should be used in pair!
137 *
138 * t -- lcut_t*
139 * s -- logical test description
140 * setup -- setup fixture function
141 * teardown -- teardown fixture function
142 */
143#define LCUT_TEST_BEGIN(t, s, setup, teardown) \
144 (t)->cut_test = NULL; \
145 (t)->cut_result = TEST_CASE_SUCCESS; \
146 if (((t)->cut_status = lcut_test_init(&(t)->cut_test, (s), (setup), (teardown))) != 0) { \
147 printf("[LCUT]: test init failed!, errcode[%d]\n", (t)->cut_status); \
148 exit(1); \
149 }
150
151#define LCUT_TEST_END(t) do { \
152 lcut_test_destroy(&(t)->cut_test); \
153 } while(0)
154
155/*
156 * initialize a Test Suite
157 *
158 * p -- lcut_ts_t*
159 * s -- suite description
160 * setup -- setup fixture function
161 * teardown -- teardown fixture function
162 */
163#define LCUT_TS_INIT(t, p, s, setup, teardown) do { \
164 if (((t)->cut_status = lcut_ts_init(&(p), (s), (setup), (teardown))) != 0) { \
165 printf("[LCUT]: test suite init failed!, errcode[%d]\n", (t)->cut_status); \
166 exit(1); \
167 } \
168 } while(0)
169
170/*
171 * Add a Test Suite to a logical Test
172 *
173 * p -- lcut_ts_t*
174 *
175 * !notice: you must not add a test case into a test suite
176 * which has already been added into a logical test!
177 */
178#define LCUT_TS_ADD(t, p) do { \
179 lcut_ts_add((t)->cut_test, (p)); \
180 } while(0)
181
182/*
183 * Add a test case to a test suite
184 *
185 * p -- lcut_ts_t*
186 * s -- test case description
187 * f -- test case function
188 * e -- extra parameter
189 */
190#define LCUT_TC_ADD(t, p, s, f, e, before, after) do { \
191 if (((t)->cut_status = lcut_tc_add((p), (s), (f), (e), (before), (after))) != 0) { \
192 printf("[LCUT]: test case add failed!, errcode[%d]\n", (t)->cut_status); \
193 exit(1); \
194 } \
195 } while(0)
196
197/*
198 * Run a logical unit test
199 */
200#define LCUT_TEST_RUN(t) do { \
201 lcut_test_run((t)->cut_test, &(t)->cut_result); \
202 } while(0)
203
204/*
205 * Report the result of the logical test
206 */
207#define LCUT_TEST_REPORT(t) do { \
208 lcut_test_report((t)->cut_test); \
209 } while(0)

211/*
212 * Get the result of a logical test
213 *
214 * in order to integrated with a Continuous integration tool, e.g. cruisecontrol.rb
215 */
216#define LCUT_TEST_RESULT(t) do { \
217 exit((t)->cut_result); \
218 } while(0)
