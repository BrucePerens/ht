133static bool
134sa818_command(
135 radio_module * const c,
136 const char * const command,
137 const char * const response,
138 returned_string /*@null@*/ /*@shared@*/ * const result)
139{
140 sa818_module * const s = c->device.sa818;
141
142 const size_t command_length = strlen(command);
143
145 if ( s->buffer == 0 )
146 return false;
147
148 if ( (*(s->platform->write))(s->platform, command, command_length) == (ssize_t)command_length ) {
149 const size_t response_length = strlen(response);
150 const ssize_t size = (*(s->platform->read))(s->platform, s->buffer, (size_t)s->buffer_size - 1);
151 if ( size >= (ssize_t)response_length ) {
152 if ( memcmp(s->buffer, response, response_length) == 0 ) {
153 if ( result ) {
154 s->buffer[size] = '\0';
155 *result = &(s->buffer[response_length]);
156 }
157 return true;
158 }
159 else
160 c->error_message = "The radio module indicated failure.";
161 }
162 else
163 c->error_message = "Read failed.";
164 }
165 else
166 c->error_message = "Write failed.";
167
168 return false;
169}

198// Return the RSSI, in dB, for the argument frequency.
199// SA-818 has a primitive scanning function, which just tells you if a frequency
200// is occupied or not. So, the returned "RSSI" will either be 0 or 128.
201static bool
202sa818_frequency_rssi(radio_module * const c, const float frequency, float * const rssi)
203{
204 char buffer[50];
205
206 (void) snprintf(buffer, sizeof(buffer), "S+%3.4f\r\n", frequency);
207 const char * result = 0;
208 if ( sa818_command(c, buffer, scan_response, &result) ) {
209 if ( !!result && *result == '0' )
210 *rssi = 255.0;
211 else
212 *rssi = 0.0;
213 return true;
214 }
215 return false;
216}

268static bool
269sa818_set(
270 radio_module * const c,
271 const radio_channel_data * const p,
272 const unsigned int channel)
273{
274 sa818_module * const s = c->device.sa818;
275 radio_channel_data * const o = &(c->device.sa818->channels[channel]);
276
277 if ( !float_equal(p->bandwidth, o->bandwidth)
278 || !float_equal(p->transmit_frequency, o->transmit_frequency)
279 || !float_equal(p->receive_frequency, o->receive_frequency)
280 || !float_equal(p->transmit_subaudible_tone, o->transmit_subaudible_tone)
281 || !float_equal(p->receive_subaudible_tone, o->receive_subaudible_tone)
282 || p->transmit_digital_code != o->transmit_digital_code
283 || p->receive_digital_code != o->receive_digital_code
284 || !float_equal(p->squelch_level, o->squelch_level) ) {
285 // Take a look at how other people encode the digital squelch code.
286 char * receive_subaudio_or_code = "0000";
287 char * transmit_subaudio_or_code = "0000";
288
289 (void) snprintf(
290 s->buffer,
291 (size_t)s->buffer_size,
292 "AT+DMOSETGROUP=%d,%3.4f,%3.4f,%s,%d,%s\r\n",
293 (int)float_equal(p->bandwidth, 25.0),
294 p->transmit_frequency,
295 p->receive_frequency,
296 receive_subaudio_or_code,
297 (int)roundf(p->squelch_level * 8.0f),
298 transmit_subaudio_or_code);
299
300 if ( !sa818_command(c, s->buffer, setgroup_response, 0) )
301 return false;
302
303 }
304
305 if ( p->low_pass_filter != o->low_pass_filter
306 || p->high_pass_filter != o->high_pass_filter
307 || p->preemphasis_deemphasis != o->preemphasis_deemphasis ) {
308 (void) snprintf(
309 s->buffer,
310 (size_t)s->buffer_size,
311 "AT+DMOSETFILTER=%d,%d,%d\r\n",
312 (int)!p->preemphasis_deemphasis,
313 (int)!p->high_pass_filter,
314 (int)!p->low_pass_filter);
315
316 if ( !sa818_command(c, s->buffer, setfilter_response, 0) )
317 return false;
318 }
319
320 if ( p->tail_tone != o->tail_tone ) {
321 (void) snprintf(
322 s->buffer,
323 (size_t)s->buffer_size,
324 "AT+DMOSETTAIL=%d\r\n",
325 (int)!p->tail_tone);
326
327 if ( !sa818_command(c, s->buffer, settail_response, 0) )
328 return false;
329 }
330
331 // Store the current radio settings.
332 memcpy(o, p, sizeof(*o));
333
334 return true;
335}

348radio_module /*@null@*/ *
349sa818(platform_context * const platform)
350{
351 // Set up the device-dependent context.
352 /*@partial@*/ radio_module * const c = malloc(sizeof(*c));
353 if ( c == 0 )
354 return 0;
355 memset(c, 0, sizeof(*c));
356 /*@partial@*/ sa818_module * const s = c->device.sa818 = malloc(sizeof(*s));
357 if ( s == 0 ) {
358 free(c);
359 return 0;
360 }
361 memset(c->device.sa818, 0, sizeof(*c->device.sa818));
362 s->platform = platform;
363
364 // Fill in the call table to provide device-dependent actions for
365 // device-independent interfaces.
366 c->channel = sa818_channel;
367 c->end = sa818_end;
368 c->frequency_rssi = sa818_frequency_rssi;
369 c->get = sa818_get;
370 c->heartbeat = sa818_heartbeat;
371 c->receive = sa818_receive;
372 c->rssi = sa818_rssi;
373 c->set = sa818_set;
374 c->transmit = sa818_transmit;
375 c->number_of_bands = 1;
376 c->number_of_subaudible_tones = (unsigned int)(sizeof(tones) / sizeof(*tones));
377 c->subaudible_tones = tones;
378 c->number_of_digital_codes = (unsigned int)(sizeof(digital_codes) / sizeof(*digital_codes));
379 c->digital_codes = digital_codes;
380 radio_band_limits * const band_limits = c->band_limits = malloc(sizeof(radio_band_limits) * c->number_of_bands);
381 if ( band_limits == 0 ) {
382 free(c->device.sa818);
383 free(c->band_limits);
384 free(c);
385 return 0;
386 }
387 memset(c->band_limits, 0, sizeof(*c->band_limits));
388 c->number_of_channels = 1;
389
390 if ( !(*(s->platform->gpio))(s->platform, SA818_ENABLE_BIT|SA818_PTT_BIT|SA818_HIGH_POWER_BIT) ) {
391 free(c->device.sa818);
392 free(c->band_limits);
393 free(c);
394 return 0;
395 }
396
397 // Delay 500 miliseconds for the radio to power up.
398 (*(s->platform->wait))(s->platform, 0.5f);
399
400 if ( sa818_command(c, connect_command, connect_response, 0) ) {
401 const char /*@observer@*/ * result = 0;
402
403 if ( sa818_command(c, version_command, version_response, &result)
404 && !!result ) {
405 c->device_name = strdup(result);
406 if ( strcmp(c->device_name, sa868_name) == 0 ) {
407 // The SA-868 has 16 channels.
408 c->number_of_channels = 16;
409 }
410 }
411 else {
412 // The device doesn't tell us its version. The SA-808 doesn't know how.
413 s->version[0] = '\0';
414 // Because I duplicate the string if the device actually returns it, I
415 // also do it here.
416 c->device_name = strdup(sa808_name);
417 }
418
419 s->channels = malloc(sizeof(radio_channel_data) * c->number_of_channels);
420 if ( s->channels == 0 ) {
421 free(c->device.sa818);
422 free(c->band_limits);
423 free(c);
424 return 0;
425 }
426 // Buffer size must be larger than the largest command or response, on SA-868
427 // we set 16 channels at once.
428 s->buffer_size = 100 + (20 * (ssize_t)c->number_of_channels);
429 s->buffer = malloc((size_t)s->buffer_size);
430 if ( s->buffer == 0 ) {
431 free(s->channels);
432 free(c->device.sa818);
433 free(c->band_limits);
434 free(c);
435 return 0;
436 }
437 memset(s->buffer, 0, (size_t)s->buffer_size);
438
439 // Poll for band limits. I don't know if this will work or what are the actual
440 // lowest and highest frequencies that can be set.
441 float rssi = 0;
442 if ( sa818_frequency_rssi(c, 134.0, &rssi) ) {
443 band_limits[0].low = 134.0;
444 band_limits[0].high = 174.0;
445 }
446 else if ( sa818_frequency_rssi(c, 400.0, &rssi) ) {
447 band_limits[0].low = 400.0;
448 band_limits[0].high = 480.0;
449 }
450 else if ( sa818_frequency_rssi(c, 320.0, &rssi) ) {
451 band_limits[0].low = 320.0;
452 band_limits[0].high = 400.0;
453 }
454
455 // This is a valid return, we can use the returned radio context for other
456 // radio commands.
457 return c;
458 }
459 free(s);
460 free(c->band_limits);
461 free(c);
462 // This is an invalid return. No commands to the device are possible.
463 return 0;
464}
