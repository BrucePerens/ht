154static bool
155sa818_command(
156 radio_module * const c,
157 const char * const command,
158 const char * const response,
159 returned_string /*@null@*/ /*@shared@*/ * const result)
160{
161 sa818_module * const s = c->device.sa818;
162
163 const size_t command_length = strlen(command);
164
165 if ( (*(s->write))(s->serial_context, command, command_length) == command_length ) {
166 const size_t response_length = strlen(response);
167 const size_t size = (*(s->read))(s->serial_context, s->buffer, s->buffer_size - 1);
168 if ( size >= response_length ) {
169 if ( memcmp(s->buffer, response, response_length) == 0 ) {
170 if ( result ) {
171 s->buffer[size] = '\0';
172 *result = &(s->buffer[response_length]);
173 }
174 return true;
175 }
176 else
177 c->error_message = "The radio module indicated failure.";
178 }
179 else
180 c->error_message = "Read failed.";
181 }
182 else
183 c->error_message = "Write failed.";
184
185 return false;
186}

288static bool
289sa818_set(
290 radio_module * const c,
291 const radio_params * const p,
292 const unsigned int channel)
293{
294 sa818_module * const s = c->device.sa818;
295 radio_params * const o = &(c->device.sa818->channels[channel]);
296
297 if ( !float_equal(p->bandwidth, o->bandwidth)
298 || !float_equal(p->transmit_frequency, o->transmit_frequency)
299 || !float_equal(p->receive_frequency, o->receive_frequency)
300 || !float_equal(p->transmit_subaudible_tone, o->transmit_subaudible_tone)
301 || !float_equal(p->receive_subaudible_tone, o->receive_subaudible_tone)
302 || p->transmit_digital_code != o->transmit_digital_code
303 || p->receive_digital_code != o->receive_digital_code
304 || !float_equal(p->squelch_level, o->squelch_level) ) {
305 // Take a look at how other people encode the digital squelch code.
306 char * receive_subaudio_or_code = "0000";
307 char * transmit_subaudio_or_code = "0000";
308
309 (void) snprintf(
310 s->buffer,
311 s->buffer_size,
312 "AT+DMOSETGROUP=%d,%3.4f,%3.4f,%s,%d,%s\r\n",
313 (int)float_equal(p->bandwidth, 25.0),
314 p->transmit_frequency,
315 p->receive_frequency,
316 receive_subaudio_or_code,
317 (int)roundf(p->squelch_level * 8.0f),
318 transmit_subaudio_or_code);
319
320 if ( !sa818_command(c, s->buffer, setgroup_response, 0) )
321 return false;
322
323 }
324
325 if ( p->low_pass_filter != o->low_pass_filter
326 || p->high_pass_filter != o->high_pass_filter
327 || p->preemphasis_deemphasis != o->preemphasis_deemphasis ) {
328 (void) snprintf(
329 s->buffer,
330 s->buffer_size,
331 "AT+DMOSETFILTER=%d,%d,%d\r\n",
332 (int)!p->preemphasis_deemphasis,
333 (int)!p->high_pass_filter,
334 (int)!p->low_pass_filter);
335
336 if ( !sa818_command(c, s->buffer, setfilter_response, 0) )
337 return false;
338 }
339
340 if ( p->tail_tone != o->tail_tone ) {
341 (void) snprintf(
342 s->buffer,
343 s->buffer_size,
344 "AT+DMOSETTAIL=%d\r\n",
345 (int)!p->tail_tone);
346
347 if ( !sa818_command(c, s->buffer, settail_response, 0) )
348 return false;
349 }
350
351 // Store the current radio settings.
352 memcpy(o, p, sizeof(*o));
353

381

392 /*@partial@*/ radio_module * const c = malloc(sizeof(*c));
393 if ( c == 0 )
394 return 0;
395 memset(c, 0, sizeof(*c));
396 /*@partial@*/ sa818_module * const s = c->device.sa818 = malloc(sizeof(*s));
397 if ( s == 0 ) {
398 free(c);
399 return 0;
400 }
401 memset(c->device.sa818, 0, sizeof(*c->device.sa818));
402 s->serial_context = serial_context;
403 s->gpio = gpio;
404 s->read = read;
405 s->write = write;
406 s->wait = wait;
407
408 // Fill in the call table to provide device-dependent actions for
409 // device-independent interfaces.
410 c->channel = sa818_channel;
411 c->end = sa818_end;
412 c->frequency_rssi = sa818_frequency_rssi;
413 c->get = sa818_get;
414 c->heartbeat = sa818_heartbeat;
415 c->receive = sa818_receive;
416 c->rssi = sa818_rssi;
417 c->set = sa818_set;
418 c->transmit = sa818_transmit;
419 c->number_of_bands = 1;
420 radio_band_limits * const band_limits = c->band_limits = malloc(sizeof(radio_band_limits) * c->number_of_bands);
421 if ( band_limits == 0 ) {
422 free(c->device.sa818);
423 free(c->band_limits);
424 free(c);
425 return 0;
426 }
427 memset(c->band_limits, 0, sizeof(*c->band_limits));
428 c->number_of_channels = 1;
429
430 // Initialize the soft GPIO settings.
431 s->enable = true;
432 s->ptt = false;
433 s->high_power = true;
434
435 // Set the hardware GPIO lines.
436 if ( !(*(s->gpio))(c) ) {
437 free(c->device.sa818);
438 free(c->band_limits);
439 free(c);
440 return 0;
441 }
442
443 // Delay 500 miliseconds for the radio to power up.
444 (*(s->wait))(0.5);
445
446 if ( sa818_command(c, connect_command, connect_response, 0) ) {
447 const char /*@observer@*/ * result = 0;
448
449 if ( sa818_command(c, version_command, version_response, &result)
450 && !!result ) {
451 c->device_name = strdup(result);
452 if ( strcmp(c->device_name, sa868_name) == 0 ) {
453 // The SA-868 has 16 channels.
454 c->number_of_channels = 16;
455 }
456 }
457 else {
458 // The device doesn't tell us its version. The SA-808 doesn't know how.
459 s->version[0] = '\0';
460 // Because I duplicate the string if the device actually returns it, I
461 // also do it here.
462 c->device_name = strdup(sa808_name);
463 }
464
465 s->channels = malloc(sizeof(radio_params) * c->number_of_channels);
466 if ( s->channels == 0 ) {
467 free(c->device.sa818);
468 free(c->band_limits);
469 free(c);
470 return 0;
471 }
472 // Buffer size must be larger than the largest command or response, on SA-868
473 // we set 16 channels at once.
474 s->buffer_size = 100 + (20 * (size_t)c->number_of_channels);
475 s->buffer = malloc(s->buffer_size);
476 if ( s->buffer == 0 ) {
477 free(s->channels);
478 free(c->device.sa818);
479 free(c->band_limits);
480 free(c);
481 return 0;
482 }
483 memset(s->buffer, 0, s->buffer_size);
484
485 // Poll for band limits. I don't know if this will work or what are the actual
486 // lowest and highest frequencies that can be set.
487 float rssi = 0;
488 if ( sa818_frequency_rssi(c, 134.0, &rssi) ) {
489 band_limits[0].low = 134.0;
490 band_limits[0].high = 174.0;
491 }
492 else if ( sa818_frequency_rssi(c, 400.0, &rssi) ) {
493 band_limits[0].low = 400.0;
494 band_limits[0].high = 480.0;
495 }

498 band_limits[0].high = 400.0;
499 }
500
501 // This is a valid return, we can use the returned radio context for other
502 // radio commands.
503 return c;
504 }
505 free(s);
506 free(c->band_limits);
507 free(c);
508 // This is an invalid return. No commands to the device are possible.
