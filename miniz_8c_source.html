11{
12 int result = 0;
13 tinfl_decompressor decomp;
14 mz_uint8 *pDict = (mz_uint8 *)MZ_MALLOC(TINFL_LZ_DICT_SIZE);
15 size_t in_buf_ofs = 0, dict_ofs = 0;
16 if (!pDict)
17 return TINFL_STATUS_FAILED;
18 memset(pDict,0,TINFL_LZ_DICT_SIZE);
19 tinfl_init(&decomp);
20 for (;;)
21 {
22 size_t in_buf_size = *pIn_buf_size - in_buf_ofs, dst_buf_size = TINFL_LZ_DICT_SIZE - dict_ofs;
23 tinfl_status status = tinfl_decompress(&decomp, (const mz_uint8 *)pIn_buf + in_buf_ofs, &in_buf_size, pDict, pDict + dict_ofs, &dst_buf_size,
24 (flags & ~(TINFL_FLAG_HAS_MORE_INPUT | TINFL_FLAG_USING_NON_WRAPPING_OUTPUT_BUF)));
25 in_buf_ofs += in_buf_size;
26 if ((dst_buf_size) && (!(*pPut_buf_func)(pDict + dict_ofs, (int)dst_buf_size, pPut_buf_user)))
27 break;
28 if (status != TINFL_STATUS_HAS_MORE_OUTPUT)
29 {
30 result = (status == TINFL_STATUS_DONE);
31 break;
32 }
33 dict_ofs = (dict_ofs + dst_buf_size) & (TINFL_LZ_DICT_SIZE - 1);
34 }
35 MZ_FREE(pDict);
36 *pIn_buf_size = in_buf_ofs;
37 return result;
38}
