9// These are events that are delivered via gm_fd_register(), which is the
10// interface to a select() loop. gm_select_task() depends on this to
11// wake up the select() when there is a new file descriptor to monitor.
12// Having this also allows us to use the select loop task to handle jobs
13// (essentially, run coroutines).
14//
15// Waking up the select() for something other than I/O is greater overhead
16// than it should be, because the FreeRTOS-lwIP combination doesn't provide
17// pipes or Unix-domain sockets, and FreeRTOS queues, the FreeRTOS equivalent
18// of pipes, don't have a file descriptor to use with select(). So presently
19// this implements a server on the loopback interface to listen for
20// communication. I could write a pipe driver using the ESP Virtual Filesystem,
21// but I haven't done so yet.
22

42static void
43event_handler(int fd, void * data, bool readable, bool writable, bool exception, bool timeout)
44{
45 char buffer[128];
46 gm_event_t * event = (gm_event_t *)buffer;
47
48 int result = read(fd, buffer, 8);
49
50 if ( result != 8 ) {
51 GM_FAIL("Event header read returned %d\n", result);
52 return;
53 }
54
55 if ( event->size > 0 ) {
56 if ( event->size > sizeof(buffer) - 8 ) {
57 GM_FAIL("size of event data too large: %d\n", event->size);
58 return;
59 }
60 result = read(fd, &event->data, event->size);
61
62 if ( result != event->size ) {
63 GM_FAIL("Event data read returned %d, error: %s\n", result, strerror(errno));
64 return;
65 }
66
67 }
68 switch ( event->operation ) {
69 case GM_EVENT_INVALID:
70 default:
71 GM_FAIL("Invalid event opcode %d\n", result);
72 break;
73 case GM_EVENT_WAKE_SELECT:
74 // If we get here, the select has already awakened, there's no need to do any more.
75 break;
76 case GM_EVENT_RUN:
77 (event->data.run.procedure)(event->data.run.data);
78 break;
79 }
