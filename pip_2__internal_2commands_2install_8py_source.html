573 def _handle_target_dir(
574 self, target_dir: str, target_temp_dir: TempDirectory, upgrade: bool
575 ) -> None:
576 ensure_dir(target_dir)
577
578 # Checking both purelib and platlib directories for installed
579 # packages to be moved to target directory
580 lib_dir_list = []
581
582 # Checking both purelib and platlib directories for installed
583 # packages to be moved to target directory
584 scheme = get_scheme("", home=target_temp_dir.path)
585 purelib_dir = scheme.purelib
586 platlib_dir = scheme.platlib
587 data_dir = scheme.data
588
589 if os.path.exists(purelib_dir):
590 lib_dir_list.append(purelib_dir)
591 if os.path.exists(platlib_dir) and platlib_dir != purelib_dir:
592 lib_dir_list.append(platlib_dir)
593 if os.path.exists(data_dir):
594 lib_dir_list.append(data_dir)
595
596 for lib_dir in lib_dir_list:
597 for item in os.listdir(lib_dir):
598 if lib_dir == data_dir:
599 ddir = os.path.join(data_dir, item)
600 if any(s.startswith(ddir) for s in lib_dir_list[:-1]):
601 continue
602 target_item_dir = os.path.join(target_dir, item)
603 if os.path.exists(target_item_dir):
604 if not upgrade:
605 logger.warning(
606 "Target directory %s already exists. Specify "
607 "--upgrade to force replacement.",
608 target_item_dir,
609 )
610 continue
611 if os.path.islink(target_item_dir):
612 logger.warning(
613 "Target directory %s already exists and is "
614 "a link. pip will not automatically replace "
615 "links, please remove if replacement is "
616 "desired.",
617 target_item_dir,
618 )
619 continue
620 if os.path.isdir(target_item_dir):
621 shutil.rmtree(target_item_dir)
622 else:
623 os.remove(target_item_dir)
624
625 shutil.move(os.path.join(lib_dir, item), target_item_dir)

639 def _warn_about_conflicts(
640 self, conflict_details: ConflictDetails, resolver_variant: str
641 ) -> None:
642 package_set, (missing, conflicting) = conflict_details
643 if not missing and not conflicting:
644 return
645
646 parts: List[str] = []
647 if resolver_variant == "legacy":
648 parts.append(
649 "pip's legacy dependency resolver does not consider dependency "
650 "conflicts when selecting packages. This behaviour is the "
651 "source of the following dependency conflicts."
652 )
653 else:
654 assert resolver_variant == "2020-resolver"
655 parts.append(
656 "pip's dependency resolver does not currently take into account "
657 "all the packages that are installed. This behaviour is the "
658 "source of the following dependency conflicts."
659 )
660
661 # NOTE: There is some duplication here, with commands/check.py
662 for project_name in missing:
663 version = package_set[project_name][0]
664 for dependency in missing[project_name]:
665 message = (
666 "{name} {version} requires {requirement}, "
667 "which is not installed."
668 ).format(
669 name=project_name,
670 version=version,
671 requirement=dependency[1],
672 )
673 parts.append(message)
674
675 for project_name in conflicting:
676 version = package_set[project_name][0]
677 for dep_name, dep_version, req in conflicting[project_name]:
678 message = (
679 "{name} {version} requires {requirement}, but {you} have "
680 "{dep_name} {dep_version} which is incompatible."
681 ).format(
682 name=project_name,
683 version=version,
684 requirement=req,
685 dep_name=dep_name,
686 dep_version=dep_version,
687 you=("you" if resolver_variant == "2020-resolver" else "you'll"),
688 )
689 parts.append(message)
690
691 logger.critical("\n".join(parts))

719def decide_user_install(
720 use_user_site: Optional[bool],
721 prefix_path: Optional[str] = None,
722 target_dir: Optional[str] = None,
723 root_path: Optional[str] = None,
724 isolated_mode: bool = False,
725) -> bool:
726 """Determine whether to do a user install based on the input options.
727
728 If use_user_site is False, no additional checks are done.
729 If use_user_site is True, it is checked for compatibility with other
730 options.
731 If use_user_site is None, the default behaviour depends on the environment,
732 which is provided by the other arguments.
733 """
734 # In some cases (config from tox), use_user_site can be set to an integer
735 # rather than a bool, which 'use_user_site is False' wouldn't catch.
736 if (use_user_site is not None) and (not use_user_site):
737 logger.debug("Non-user install by explicit request")
738 return False
739
740 if use_user_site:
741 if prefix_path:
742 raise CommandError(
743 "Can not combine '--user' and '--prefix' as they imply "
744 "different installation locations"
745 )
746 if virtualenv_no_global():
747 raise InstallationError(
748 "Can not perform a '--user' install. User site-packages "
749 "are not visible in this virtualenv."
750 )
751 logger.debug("User install by explicit request")
752 return True
753
754 # If we are here, user installs have not been explicitly requested/avoided
755 assert use_user_site is None
756
757 # user install incompatible with --prefix/--target
758 if prefix_path or target_dir:
759 logger.debug("Non-user install due to --prefix or --target option")
760 return False
761
762 # If user installs are not enabled, choose a non-user install
763 if not site.ENABLE_USER_SITE:
764 logger.debug("Non-user install because user site-packages disabled")
765 return False
766
767 # If we have permission for a non-user install, do that,
768 # otherwise do a user install.
769 if site_packages_writable(root=root_path, isolated=isolated_mode):
770 logger.debug("Non-user install because site-packages writeable")
771 return False
772
773 logger.info(
774 "Defaulting to user installation because normal site-packages "
775 "is not writeable"
776 )

780def reject_location_related_install_options(
781 requirements: List[InstallRequirement], options: Optional[List[str]]
782) -> None:
783 """If any location-changing --install-option arguments were passed for
784 requirements or on the command-line, then show a deprecation warning.
785 """
786
787 def format_options(option_names: Iterable[str]) -> List[str]:
788 return ["--{}".format(name.replace("_", "-")) for name in option_names]
789
790 offenders = []
791
792 for requirement in requirements:
793 install_options = requirement.install_options
794 location_options = parse_distutils_args(install_options)
795 if location_options:
796 offenders.append(
797 "{!r} from {}".format(
798 format_options(location_options.keys()), requirement
799 )
800 )
801
802 if options:
803 location_options = parse_distutils_args(options)
804 if location_options:
805 offenders.append(
806 "{!r} from command line".format(format_options(location_options.keys()))
807 )
808
809 if not offenders:
810 return
811
812 raise CommandError(
813 "Location-changing options found in --install-option: {}."
814 " This is unsupported, use pip-level options like --user,"
815 " --prefix, --root, and --target instead.".format("; ".join(offenders))
816 )

819def create_os_error_message(
820 error: OSError, show_traceback: bool, using_user_site: bool
821) -> str:
822 """Format an error message for an OSError
823
824 It may occur anytime during the execution of the install command.
825 """
826 parts = []
827
828 # Mention the error if we are not going to show a traceback
829 parts.append("Could not install packages due to an OSError")
830 if not show_traceback:
831 parts.append(": ")
832 parts.append(str(error))
833 else:
834 parts.append(".")
835
836 # Spilt the error indication from a helper message (if any)
837 parts[-1] += "\n"
838
839 # Suggest useful actions to the user:
840 # (1) using user site-packages or (2) verifying the permissions
841 if error.errno == errno.EACCES:
842 user_option_part = "Consider using the `--user` option"
843 permissions_part = "Check the permissions"
844
845 if not running_under_virtualenv() and not using_user_site:
846 parts.extend(
847 [
848 user_option_part,
849 " or ",
850 permissions_part.lower(),
851 ]
852 )
853 else:
854 parts.append(permissions_part)
855 parts.append(".\n")
856
857 # Suggest the user to enable Long Paths if path length is
858 # more than 260
859 if (
860 WINDOWS
861 and error.errno == errno.ENOENT
862 and error.filename
863 and len(error.filename) > 260
864 ):
865 parts.append(
866 "HINT: This error might have occurred since "
867 "this system does not have Windows Long Path "
868 "support enabled. You can find information on "
869 "how to enable this at "
870 "https://pip.pypa.io/warnings/enable-long-paths\n"
871 )
872
873 return "".join(parts).strip() + "\n"
