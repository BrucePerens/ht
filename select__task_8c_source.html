101void
102gm_fd_unregister(const int fd) {
103 handlers[fd] = (gm_fd_handler_t)0;
104 data[fd] = 0;
105 timerclear(&expirations[fd]);
106 timerclear(&intervals[fd]);
107 FD_CLR(fd, &read_fds);
108 FD_CLR(fd, &write_fds);
109 FD_CLR(fd, &exception_fds);
110 FD_CLR(fd, &monitored_fds);
111
112 // If the last FD is cleared, set fd_limit to the highest remaining set fd,
113 // plus one.
114 if ( fd_limit == fd + 1 ) {
115 fd_limit = 0;
116 for ( int i = fd - 1; i >= 0; i-- ) {
117 if ( FD_ISSET(i, &monitored_fds) ) {
118 fd_limit = i + 1;
119 break;
120 }
121 }
122 }
123 if ( in_select ) {
124 // Select is waiting on an FD that is about to not be valid.
125 // Wake it up and restart it without that FD.
126 gm_select_wakeup();
127 }
128}

130static void
131select_task(void * param)
132{
133 for ( ; ; ) {
134 fd_set read_now;
135 fd_set write_now;
136 fd_set exception_now;
137
138 memcpy(&read_now, &read_fds, sizeof(read_now));
139 memcpy(&write_now, &write_fds, sizeof(write_now));
140 memcpy(&exception_now, &exception_fds, sizeof(exception_now));
141
142 struct timeval time_of_day_before_select;
143 gettimeofday(&time_of_day_before_select, 0);
144
145 struct timeval select_blocking_interval = {};
146 select_blocking_interval.tv_sec = 1 << 30;
147
148
149 for ( int i = 0; i < fd_limit; i++ ) {
150 if ( FD_ISSET(i, &monitored_fds) ) {
151 // Check if the FD was closed without being unregistered.
152 if ( lseek(i, 0, SEEK_CUR) < 0 && errno == EBADF ) {
153 GM_FAIL("select_task(): fd %d isn't an open file descriptor and gm_fd_unregister() wasn't called.\n");
154 gm_fd_unregister(i);
155 }
156
157 const struct timeval * const t = &expirations[i];
158
159 if ( timerisset(t) ) {
160 if ( timercmp(t, &time_of_day_before_select, <) ) {
161 timerclear(&select_blocking_interval);
162 break; // Can't get lower than this, so no point in checking more values.
163 }
164 else {
165 struct timeval interval;
166 timersub(t, &time_of_day_before_select, &interval);
167
168 if ( timercmp(&interval, &select_blocking_interval, <) )
169 select_blocking_interval = interval;
170 }
171 }
172 }
173 }
174
175 in_select = true;
176 const int number_of_set_fds = select(
177 fd_limit,
178 &read_now,
179 &write_now,
180 &exception_now,
181 &select_blocking_interval);
182 in_select = false;
183
184 if ( number_of_set_fds < 0 ) {
185 // Select failed.
186 if ( errno == EBADF ) {
187 // A file descriptor was closed while select() was sleeping upon it.
188 // FIX: Check that the lseek() call above detects bad sockets. Or this
189 // might become a non-terminating loop.
190 continue;
191 }
192 else
193 GM_FAIL("Select failed");
194 }
195 struct timeval time_of_day_after_select;
196 gettimeofday(&time_of_day_after_select, 0);
197
198 for ( unsigned int i = 0; i < fd_limit; i++ ) {
199 if ( FD_ISSET(i, &monitored_fds) ) {
200 const bool readable = FD_ISSET(i, &read_now);
201 const bool writable = FD_ISSET(i, &write_now);
202 const bool exception = FD_ISSET(i, &exception_now);
203 bool timeout = false;
204
205 struct timeval * t = &expirations[i];
206 if ( timerisset(t) ) {
207 if ( timercmp(t, &time_of_day_after_select, <) )
208 timeout = true;
209 else {
210 // Compensate for timer granularity, rather than looping through
211 // select() until the timer runs out.
212 struct timeval remaining;
213
214 timersub(t, &time_of_day_after_select, &remaining);
215 if ( remaining.tv_sec == 0 && remaining.tv_usec < 10000 )
216 timeout = true;
217 }
218 }
219
220 if ( readable || writable || exception || timeout ) {
221 gm_fd_handler_t handler = handlers[i];
222
223 if ( readable || writable || exception )
224 timeout = false;
225
226 timeradd(
227 &intervals[i],
228 &time_of_day_after_select,
229 &expirations[i]);
230
231 if (handler)
232 (handler)(i, data[i], readable, writable, exception, timeout);
233 }
234 }
235 }
236 }
237}
