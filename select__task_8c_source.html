55gm_fd_register(int fd, gm_fd_handler_t handler, void * d, bool readable, bool writable, bool exception, uint32_t seconds) {
56 int limit = fd + 1;
57 struct timeval now;
58 struct timeval * t;
59
60 if ( fd_limit < limit )
61 fd_limit = limit;
62
63 data[fd] = d;
64
65 gettimeofday(&now, 0);
66
67 t = &timeouts[fd];
68
69 if ( seconds ) {
70 t->tv_sec = now.tv_sec + seconds;
71 t->tv_usec = now.tv_usec;
72 }
73 else {
74 timerclear(t);
75 }
76
77 if ( readable )
78 FD_SET(fd, &read_fds);
79 else
80 FD_CLR(fd, &read_fds);
81
82 if ( writable )
83 FD_SET(fd, &write_fds);
84 else
85 FD_CLR(fd, &write_fds);
86
87 if ( exception )
88 FD_SET(fd, &exception_fds);
89 else

93
94 if ( in_select )
95 gm_select_wakeup();

98void
99gm_fd_unregister(int fd) {
100 handlers[fd] = (gm_fd_handler_t)0;
101 data[fd] = 0;
102 timerclear(&timeouts[fd]);
103 FD_CLR(fd, &read_fds);
104 FD_CLR(fd, &write_fds);
105 FD_CLR(fd, &exception_fds);
106
107 // If the last FD is cleared, set fd_limit to the highest remaining set fd, plus one.
108 if ( fd_limit == fd + 1 ) {
109 fd_limit = 0;
110 for ( int i = fd - 1; i >= 0; i-- ) {
111 if ( FD_ISSET(i, &read_fds) || FD_ISSET(i, &write_fds) || FD_ISSET(i, &exception_fds) ) {
112 fd_limit = i + 1;
113 break;
114 }
115 }
116 }
117 if ( in_select )
118 gm_select_wakeup();
119}

121static void
122select_task(void * param)
123{
124 for ( ; ; ) {
125 fd_set read_now;
126 fd_set write_now;
127 fd_set exception_now;
128 fd_set monitored_fds = {};
129 struct timeval now;
130 struct timeval min_time = { 365 * 24 * 60 * 60, 0 }; // Absurdly long time.
131 struct timeval when;
132 int number_of_set_fds;
133
134 in_select = true;
135
136 memcpy(&read_now, &read_fds, sizeof(read_now));
137 memcpy(&write_now, &write_fds, sizeof(write_now));
138 memcpy(&exception_now, &exception_fds, sizeof(exception_now));
139
140 gettimeofday(&now, 0);
141
142 for ( unsigned int i = 0; i < fd_limit; i++ ) {
143 if ( FD_ISSET(i, &read_now) || FD_ISSET(i, &write_now) || FD_ISSET(i, &exception_now) ) {
144
145 if ( lseek(i, 0, SEEK_CUR) < 0 && errno == EBADF ) {
146 gm_printf("select_task(): fd %d isn't an open file descriptor.\n");
147 continue;
148 }
149 FD_SET(i, &monitored_fds);
150
151 struct timeval * t = &timeouts[i];
152
153 if ( timerisset(t) ) {
154
155 if ( timercmp(t, &now, <) ) {
156 timerclear(&min_time);
157 break; // Can't get lower than this, so no point in checking more values.
158 }
159 else
160 timersub(t, &now, &when);
161
162 if ( timercmp(&when, &min_time, <) )
163 min_time = when;
164 }
165 }
166 }
167 for ( unsigned int i = 0; i < timer_task_limit; i++ ) {
168 struct timer_task * t = &timer_tasks[i];
169 if ( t->procedure ) {
170 if ( timercmp(&t->when, &now, >) ) {
171 (t->procedure)(t->data);
172 timerclear(&t->when);
173 t->procedure = 0;
174 t->data = 0;
175 }
176 else {
177 timersub(&now, &t->when, &when);
178
179 if ( timercmp(&when, &min_time, <) )
180 min_time = when;
181 }
182 }
183 }
184
185 number_of_set_fds = select(fd_limit, &read_now, &write_now, &exception_now, &min_time);
186 in_select = false;
187
188 if ( number_of_set_fds > 0 ) {
189 for ( unsigned int i = 0; i < fd_limit; i++ ) {
190 if ( FD_ISSET(i, &monitored_fds) ) {
191 struct timeval * t = &timeouts[i];
192 bool readable = false;
193 bool writable = false;
194 bool exception = false;
195 bool timeout = false;
196
197 if ( FD_ISSET(i, &read_now) )
198 readable = true;
199 if ( FD_ISSET(i, &write_now) )
200 writable = true;
201 if ( FD_ISSET(i, &exception_now) )
202 exception = true;
203 if ( timerisset(t) ) {
204 if ( timercmp(t, &now, <) )
205 timeout = true;
206 else {
207 // Compensate for timer granularity, rather than looping through select() until the
208 // timer runs out.
209 struct timeval remaining;
210
211 timersub(t, &now, &remaining);
212 if ( remaining.tv_sec == 0 && remaining.tv_usec < 10000 )
213 timeout = true;
214 }
215 }
216
217 if ( readable || writable || exception || timeout ) {
218 // Save the handler and data before unregister. Call unregister _before_ calling the handler,
219 // which may call register for the same FD.
220 gm_fd_handler_t handler = handlers[i];
221 void * d = data[i];
222
223 if ( timeout )
224 gm_fd_unregister(i);
225 if (handler)
226 (handler)(i, d, readable, writable, exception, timeout);
227 }
228
229 }
230 }
231 }
232 else if ( number_of_set_fds == 0 ) {
233 bool timeout = false;
234
235 for ( unsigned int i = 0; i < fd_limit; i++ ) {
236 if ( FD_ISSET(i, &monitored_fds) ) {
237 struct timeval * t = &timeouts[i];
238 if ( timerisset(t) ) {
239 if ( timercmp(t, &now, <) )
240 timeout = true;
241 else {
242 // Compensate for timer granularity, rather than looping through select() until the
243 // timer runs out.
244 struct timeval remaining;
245
246 timersub(t, &now, &remaining);
247 if ( remaining.tv_sec == 0 && remaining.tv_usec < 10000 )
248 timeout = true;
249 }
250 }
251 if ( timeout ) {
252 // Save the handler and data before unregister. Call unregister _before_ calling the handler,
253 // which may call register for the same FD.
254 gm_fd_handler_t handler = handlers[i];
255 void * d = data[i];
256
257 gm_fd_unregister(i);
258 if (handler)
259 (handler)(i, d, false, false, false, true);
260 }
261 }
262 }
263 }
264 else {
265 // Select failed.
266 if ( errno == EBADF ) {
267 // A file descriptor was closed while select() was running upon it.
268 }
269 else
270 GM_FAIL("Select failed");
271 }
272 }
