15// We use JSON for session context. It gets sent to the browser as an encrypted
16// cookie, and the browser sends it back. On a machine with more resources, we
17// would keep the session context on a disk, or at least a larger FLASH with
18// better wear-leveling. ESP-32 has small FLASH with a limited number of write
19// cycles, and its wear-leveling depends on the presence of _unused_ space on the
20// partition, which of course is limited by the FLASH size. Once you ruin the
21// FLASH the device is probably bricked. So don't store session context locally.

40// This buffer gets used for two different things in write_cookie(), to save stack
41// on the very memory-constrained ESP-32.
42union CookiePayload {
43 // Struct payload contains compact JSON, with a header.
44 struct Payload {
45 struct Header {
46 // This checksum is used to verify that we have decrypted the payload
47 // data successfully, and to prevent bit-manipulation of the
48 // cookie as a means of breaking our encryption.
49 uint32_t crc32;
50
51 // Cookie encryption always uses the same AES initialization vector,
52 // so that we don't have to send the IV as a separate cookie.
53 // Thus, we _must_ include randomness in the payload so that the
54 // encrypted output is not the same every time for the same JSON input,
55 // which would provide information for breaking our encryption.
56 uint32_t random;
57 } header;
58
59 char data[roundup(PAYLOAD_SIZE + sizeof(struct Header), 16)];
60 } payload;
61
62 // Struct Cookie contains the Set-Cookie datum.
63 struct Cookie {
64 // This contains cookie_start[] without the terminating null.
65 uint8_t start[sizeof(cookie_start) - 1];
66
67 // This contains the base64-encoded encrypted JSON data.
68 uint8_t data[
69 sizeof(struct Payload)
70 + (sizeof(struct Payload) / 3) + 3 // Account for Base64-encoding.
71 + sizeof(cookie_end)];
72 } cookie;
73
74 // Typed access to encrypt struct Payload.
75 uint8_t bytes[sizeof(struct Cookie)];
76
77 // Typed access to send struct Cookie as the Set-Cookie datum.

81cJSON *
82gm_read_cookie(httpd_req_t * req)
83{
84 union CookiePayload b;
85 char a[sizeof(b.characters)];
86 size_t data_length = sizeof(a);
87
88 esp_err_t cookie_err = httpd_req_get_cookie_val(req, "c", (char *)a, &data_length);
89
90 // The most likely error here is simply that the current request did not
91 // contain our session data cookie.
92 if ( cookie_err == ESP_ERR_NOT_FOUND )
93 return 0; // Fail quietly.
94 else if ( cookie_err != ESP_OK ) {
95 gm_printf("Cookie error: %s.\n", esp_err_to_name(cookie_err));
96 return 0;
97 }
98
99 // Decode from Base-64 and get the encrypted payload.
100 size_t length = 0;
101 uint8_t * const unbased = base64_decode(a, data_length, &length);
102 memcpy(a, unbased, length);
103 a[length] = '\0';
104 free(unbased);
105
106 if ( length % 16 > 0 ) {
107 gm_printf("Cookie payload length %d is not a multiple of 16.\n", length);
108 return 0;
109 }
110
111 // AES modifies the IV so that you can use it to stream. So, make a temporary
112 // copy.
113 uint8_t iv[sizeof(GM.aes_cookie_iv)];
114 memcpy(iv, GM.aes_cookie_iv, sizeof(iv));
115
116 // Decrypt the payload.
117 (void) esp_aes_crypt_cbc(
118 &GM.aes_cookie_context,
119 ESP_AES_DECRYPT,
120 length,
121 iv,
122 (uint8_t *)a,
123 b.bytes);
124
125 // Check that the calculated CRC for the data matches the one in the cookie.
126 const uint32_t crc32 = esp_rom_crc32_le(
127 0,
128 (uint8_t *)&b.payload.header.random, length - sizeof(b.payload.header.crc32));
129
130 if ( b.payload.header.crc32 != crc32 ) {
131 gm_printf("Cookie CRC doesn't match.\n");
132 return 0;
133 }
134
135 // Parse the payload.
136 cJSON * volatile json = cJSON_ParseWithLength(
137 b.payload.data,
138 length - sizeof(b.payload.header));
139
140 if ( json == NULL ) {
141 gm_printf("Cookie JSON parse failed.\n");
142 return 0;
143 }
144
145 return json;
146}

148void
149gm_write_cookie(httpd_req_t * const req, cJSON * const json)
150{
151 // Allow for a 1024 bytes of compact JSON. This effects the necessary httpd
152 // header size, and the httpsd stack size.
153 // The http to https redirector doesn't use cookies, and these cookies are only
154 // sent to https connections.
155 // The + 8 is to account for the size of the header in struct Payload.
156 uint8_t a[PAYLOAD_SIZE + sizeof(struct Header)];
157 union CookiePayload b;
158
159 if ( cJSON_PrintPreallocated(
160 json,
161 b.payload.data,
162 sizeof(b.payload.data),
163 0) == 0 ) {
164 ESP_LOGE(tag, "JSON context too large for buffer.\n");
165 sleep(1);
166 abort();
167 }
168
169 const size_t length = strlen((const char *)b.payload.data);
170
171 // Because we always use the same initialization vector for AES,
172 // add randomness to make the plaintext unpredictable. This avoids
173 // having to send the initialization vector in another cookie.
174 esp_fill_random(&b.payload.header.random, sizeof(b.payload.header.random));
175
176
177 // Calculate padding, because AES will only process data of a size that
178 // is a multiple of 16.
179 const size_t padding = roundup(length + sizeof(b.payload.header), 16)
180 - (length + sizeof(b.payload.header));
181 memset(&b.payload.data[length], 0, padding);
182
183 // Add a 32-bit checksum to the encrypted content, to defeat bit manipulation
184 // and to verify successful decryption. We checksum the padding because when
185 // we decrypt the payload, it includes the padding and it's difficult to figure
186 // out the length of the padding in the read function without a size field.
187 b.payload.header.crc32 = esp_rom_crc32_le(
188 0,
189 (uint8_t *)&b.payload.header.random,
190 sizeof(b.payload.header.random) + length + padding);
191
192 // AES modifies the IV so that you can use it to stream. So, make a temporary
193 // copy.
194 uint8_t iv[sizeof(GM.aes_cookie_iv)];
195 memcpy(iv, GM.aes_cookie_iv, sizeof(iv));
196
197 // Encrypt the payload.
198 (void) esp_aes_crypt_cbc(
199 &GM.aes_cookie_context,
200 ESP_AES_ENCRYPT,
201 length + sizeof(b.payload.header) + padding,
202 iv,
203 b.bytes,
204 a);
205
206 // Base-64-encode the encrypted payload, because the Set-Cookie datum can't
207 // contain binary data or characters important to the parse like space or
208 // semicolon.
209 size_t base64_length = 0;
210 char * const encoded = base64_encode(
211 a,
212 sizeof(b.payload.header) + length + padding,
213 &base64_length);
214
215 // Construct the Set-Cookie datum.
216 memcpy(b.cookie.start, cookie_start, sizeof(b.cookie.start));
217 // base64_encode adds a newline character, don't include it.
218 base64_length--;
219 memcpy(b.cookie.data, encoded, base64_length);
220 free(encoded);
221 // This includes null-termination.
222 memcpy(&b.cookie.data[base64_length], cookie_end, sizeof(cookie_end));
223
224 httpd_resp_set_hdr(req, "Set-Cookie", b.characters);
225}
