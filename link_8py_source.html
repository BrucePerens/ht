44@dataclass(frozen=True)
45class LinkHash:
46 """Links to content may have embedded hash values. This class parses those.
47
48 `name` must be any member of `_SUPPORTED_HASHES`.
49
50 This class can be converted to and from `ArchiveInfo`. While ArchiveInfo intends to
51 be JSON-serializable to conform to PEP 610, this class contains the logic for
52 parsing a hash name and value for correctness, and then checking whether that hash
53 conforms to a schema with `.is_hash_allowed()`."""
54
55 name: str
56 value: str
57
58 _hash_re = re.compile(
59 # NB: we do not validate that the second group (.*) is a valid hex
60 # digest. Instead, we simply keep that string in this class, and then check it
61 # against Hashes when hash-checking is needed. This is easier to debug than
62 # proactively discarding an invalid hex digest, as we handle incorrect hashes
63 # and malformed hashes in the same place.
64 r"({choices})=(.*)".format(
65 choices="|".join(re.escape(hash_name) for hash_name in _SUPPORTED_HASHES)
66 ),
67 )
68
69 def __post_init__(self) -> None:
70 assert self._hash_re.match(f"{self.name}={self.value}")
71
72 @classmethod
73 @functools.lru_cache(maxsize=None)
74 def split_hash_name_and_value(cls, url: str) -> Optional["LinkHash"]:
75 """Search a string for a checksum algorithm name and encoded output value."""
76 match = cls._hash_re.search(url)
77 if match is None:
78 return None
79 name, value = match.groups()
80 return cls(name=name, value=value)
81
82 def as_dict(self) -> Dict[str, str]:
83 return {self.name: self.value}
84
85 def as_hashes(self) -> Hashes:
86 """Return a Hashes instance which checks only for the current hash."""
87 return Hashes({self.name: [self.value]})
88
89 def is_hash_allowed(self, hashes: Optional[Hashes]) -> bool:
90 """
91 Return True if the current hash is allowed by `hashes`.
92 """
93 if hashes is None:
94 return False
95 return hashes.is_hash_allowed(self.name, hex_digest=self.value)

98def _clean_url_path_part(part: str) -> str:
99 """
100 Clean a "part" of a URL path (i.e. after splitting on "@" characters).
101 """
102 # We unquote prior to quoting to make sure nothing is double quoted.
103 return urllib.parse.quote(urllib.parse.unquote(part))

106def _clean_file_url_path(part: str) -> str:
107 """
108 Clean the first part of a URL path that corresponds to a local
109 filesystem path (i.e. the first part after splitting on "@" characters).
110 """
111 # We unquote prior to quoting to make sure nothing is double quoted.
112 # Also, on Windows the path part might contain a drive letter which
113 # should not be quoted. On Linux where drive letters do not
114 # exist, the colon should be quoted. We rely on urllib.request
115 # to do the right thing here.
116 return urllib.request.pathname2url(urllib.request.url2pathname(part))

119# percent-encoded: /
120_reserved_chars_re = re.compile("(@|%2F)", re.IGNORECASE)
121
122
123def _clean_url_path(path: str, is_local_path: bool) -> str:
124 """
125 Clean the path portion of a URL.
126 """
127 if is_local_path:
128 clean_func = _clean_file_url_path
129 else:
130 clean_func = _clean_url_path_part
131
132 # Split on the reserved characters prior to cleaning so that
133 # revision strings in VCS URLs are properly preserved.
134 parts = _reserved_chars_re.split(path)
135
136 cleaned_parts = []
137 for to_clean, reserved in pairwise(itertools.chain(parts, [""])):
138 cleaned_parts.append(clean_func(to_clean))
139 # Normalize %xx escapes (e.g. %2f -> %2F)
140 cleaned_parts.append(reserved.upper())
141
142 return "".join(cleaned_parts)

145def _ensure_quoted_url(url: str) -> str:
146 """
147 Make sure a link is fully quoted.
148 For example, if ' ' occurs in the URL, it will be replaced with "%20",
149 and without double-quoting other characters.
150 """
151 # Split the URL into parts according to the general structure
152 # `scheme://netloc/path;parameters?query#fragment`.
153 result = urllib.parse.urlparse(url)
154 # If the netloc is empty, then the URL refers to a local filesystem path.
155 is_local_path = not result.netloc
156 path = _clean_url_path(result.path, is_local_path=is_local_path)
157 return urllib.parse.urlunparse(result._replace(path=path))

187 :param comes_from: instance of IndexContent where the link was found,
188 or string.
189 :param requires_python: String containing the `Requires-Python`
190 metadata field, specified in PEP 345. This may be specified by

192 described in PEP 503.
193 :param yanked_reason: the reason the file has been yanked, if the
194 file has been yanked, or None if the file hasn't been yanked.
195 This is the value of the "data-yanked" attribute, if present, in

198 PEP 592 for more information and the specification.

200 metadata is provided. This is the value of the "data-dist-info-metadata"

203 information and the specification.
204 :param cache_link_parsing: A flag that is used elsewhere to determine
205 whether resources retrieved from this link should be cached. PyPI
206 URLs should generally have this set to False, for example.
207 :param hashes: A mapping of hash names to digests to allow us to
208 determine the validity of a download.

464class _CleanResult(NamedTuple):
465 """Convert link for equivalency check.
466
467 This is used in the resolver to check whether two URL-specified requirements
468 likely point to the same distribution and can be considered equivalent. This
469 equivalency logic avoids comparing URLs literally, which can be too strict
470 (e.g. "a=1&b=2" vs "b=2&a=1") and produce conflicts unexpecting to users.
471
472 Currently this does three things:
473
474 1. Drop the basic auth part. This is technically wrong since a server can
475 serve different content based on auth, but if it does that, it is even
476 impossible to guarantee two URLs without auth are equivalent, since
477 the user can input different auth information when prompted. So the
478 practical solution is to assume the auth doesn't affect the response.
479 2. Parse the query to avoid the ordering issue. Note that ordering under the
480 same key in the query are NOT cleaned; i.e. "a=1&a=2" and "a=2&a=1" are
481 still considered different.
482 3. Explicitly drop most of the fragment part, except ``subdirectory=`` and
483 hash values, since it should have no impact the downloaded content. Note
484 that this drops the "egg=" part historically used to denote the requested
485 project (and extras), which is wrong in the strictest sense, but too many
486 people are supplying it inconsistently to cause superfluous resolution
487 conflicts, so we choose to also ignore them.
488 """
489
490 parsed: urllib.parse.SplitResult
491 query: Dict[str, List[str]]
492 subdirectory: str
493 hashes: Dict[str, str]
494
495
496def _clean_link(link: Link) -> _CleanResult:
497 parsed = link._parsed_url
498 netloc = parsed.netloc.rsplit("@", 1)[-1]
499 # According to RFC 8089, an empty host in file: means localhost.
500 if parsed.scheme == "file" and not netloc:
501 netloc = "localhost"
502 fragment = urllib.parse.parse_qs(parsed.fragment)
503 if "egg" in fragment:
504 logger.debug("Ignoring egg= fragment in %s", link)
505 try:
506 # If there are multiple subdirectory values, use the first one.
507 # This matches the behavior of Link.subdirectory_fragment.
508 subdirectory = fragment["subdirectory"][0]
509 except (IndexError, KeyError):
510 subdirectory = ""
511 # If there are multiple hash values under the same algorithm, use the
512 # first one. This matches the behavior of Link.hash_value.
513 hashes = {k: fragment[k][0] for k in _SUPPORTED_HASHES if k in fragment}
514 return _CleanResult(
515 parsed=parsed._replace(netloc=netloc, query="", fragment=""),
516 query=urllib.parse.parse_qs(parsed.query),
517 subdirectory=subdirectory,
518 hashes=hashes,
519 )
