130static int
131request_mapping_ipv4(gm_port_mapping_t * m)
132{
133 last_request_t * r = &last_request_ipv4;
134 nat_pmp_or_pcp_t * p = &r->packet;
135 socklen_t send_address_size;
136 ssize_t send_result;
137 struct sockaddr_in * a4 = (struct sockaddr_in *)&r->address;
138
139 memset(r, '\0', sizeof(last_request_ipv4));
140 r->valid = true;
141
142 send_address_size = sizeof(struct sockaddr_in);
143 a4->sin_addr.s_addr = GM.sta.ip4.router.sin_addr.s_addr;
144 a4->sin_family = AF_INET;
145 a4->sin_port = htons(PCP_PORT);
146 p->pcp.request.client_address.s6_addr[10] = 0xff;
147 p->pcp.request.client_address.s6_addr[11] = 0xff;
148 memcpy(&p->pcp.request.client_address.s6_addr[12], &GM.sta.ip4.address.sin_addr.s_addr, 4);
149
150 // This sets the requested address to the all-zeroes IPV4 address: ::ffff:0.0.0.0 .
151 // if external_address was all zeroes, it would be the all-zeroes IPV6 address.
152 // Some hosts would take that to mean a request for an IPV6-to-IPV4 port mapping.
153 p->pcp.mp.external_address.s6_addr[10] = 0xff;
154 p->pcp.mp.external_address.s6_addr[11] = 0xff;
155
156 p->version = PORT_MAPPING_PROTOCOL;
157 p->opcode = PCP_MAP;
158 p->pcp.lifetime = htonl(24 * 60 * 60);
159 memcpy(p->pcp.mp.nonce, m->nonce, sizeof(m->nonce));
160 p->pcp.mp.protocol = m->tcp ? PCP_TCP : PCP_UDP;
161 p->pcp.mp.internal_port = htons(m->internal_port);
162 p->pcp.mp.external_port = htons(m->external_port);
163 p->pcp.lifetime = htonl(m->lifetime);
164
165 // Send the packet to the gateway.
166 send_result = sendto(
167 ipv4_unicast_sock,
168 p,
169 map_packet_size,
170 0,
171 (const struct sockaddr *)a4,
172 send_address_size);
173 gettimeofday(&r->time, 0);
174
175 if ( send_result < map_packet_size ) {
176 GM_FAIL("Send returned %d\n", send_result);
177 return -1;
178 }
179 return 0;
180}

347void
348decode_packet(nat_pmp_or_pcp_t * p, ssize_t message_size, bool multicast, struct sockaddr_storage * address)
349{
350 bool response;
351 uint16_t port;
352
353 // FIX: This assumes all messages are PCP, it should handle NAT-PMP correctly.
354
355 switch ( address->ss_family ) {
356 case AF_INET:
357 port = htons(((struct sockaddr_in *)address)->sin_port);
358 break;
359 case AF_INET6:
360 port = htons(((struct sockaddr_in6 *)address)->sin6_port);
361 break;
362 default:
363 GM_FAIL("decode_packet(): Address family %d.\n", address->ss_family);
364 return;
365 }
366
367 if ( port != PCP_PORT ) {
368 GM_FAIL("Ignoring message that isn't from the PCP port.\n");
369 return;
370 }
371
372 switch ( address->ss_family ) {
373 case AF_INET:
374 if ( ((struct sockaddr_in *)address)->sin_addr.s_addr != GM.sta.ip4.router.sin_addr.s_addr ) {
375 GM_FAIL("IPV4 packet not from the router, ignored.\n");
376 return;
377 }
378 break;
379 case AF_INET6:
380 break;
381 }
382
383
384 response = p->opcode & 0x80;
385
386 switch ( p->opcode & 0x7f ) {
387 case PCP_ANNOUNCE:
388 decode_pcp_announce(p, message_size, multicast, address);
389 break;
390 case PCP_MAP:
391 case PCP_PEER:
392 if ( !response ) {
393 GM_FAIL("Client received request.\n");
394 return;
395 }
396 if ( multicast ) {
397 GM_FAIL("Client received multicast request.\n");
398 return;
399 }
400 switch ( p->opcode & 0x7f ) {
401 case PCP_MAP:
402 if ( message_size < map_packet_size ) {
403 GM_FAIL("Receive packet too small for MAP: %d\n", message_size);
404 return;
405 }
406 decode_pcp_map(p, message_size, multicast, address);
407 break;
408 case PCP_PEER:
409 if ( message_size < sizeof(nat_pmp_or_pcp_t) ) {
410 GM_FAIL("Receive packet too small for PEER: %d\n", message_size);
411 return;
412 }
413 decode_pcp_peer(p, message_size, multicast, address);
414 break;
415 }
416 break;
417 default:
418 GM_FAIL("Unrecognized opcode %x\n", p->opcode);
419 }
420}

464 // FIX: Time-out responses and retry, eventually abandon trying.
465 int value = 1;
466 struct ip_mreq multi_request = {};

468
469 ipv4_multicast_sock = socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP);
470
471 // IPv4 "all hosts" multicast.
472 inet_pton(AF_INET, "224.0.0.1", &multi_request.imr_multiaddr);

478
479 // Reuse addresses, because other software listens for all-hosts multicast.
480 if ( setsockopt(ipv4_multicast_sock, SOL_SOCKET, SO_REUSEADDR, &value, sizeof(value)) < 0 ) {
481 GM_FAIL("Setsockopt SOL_SOCKET failed: %s.\n", strerror(errno));
482 return;
483 }
484 if ( bind(ipv4_multicast_sock, (struct sockaddr *)&address, sizeof(address)) < 0 ) {
485 GM_FAIL("bind failed.\n");
486 return;
487 }
488
489 if ( setsockopt(ipv4_multicast_sock, IPPROTO_IP, IP_ADD_MEMBERSHIP, &multi_request, sizeof(multi_request)) < 0 ) {
490 // This fails because the host is already registered to the "all-hosts" multicast
491 // group. Ignore that.
492 if ( errno != EADDRNOTAVAIL ) {
493 GM_FAIL("Setsockopt IP_ADD_MEMBERSHIP failed: %s.\n", strerror(errno));
494 return;
495 }
496 }
497 // Data is set to 1 for multicast, 0 for unicast.
498 gm_fd_register(ipv4_multicast_sock, incoming_packet, (void *)1, true, false, true, 0);
