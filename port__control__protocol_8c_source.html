347void
348decode_packet(nat_pmp_or_pcp_t * p, ssize_t message_size, struct sockaddr_storage * address)
349{
350 bool response;
351 uint16_t port;
352
353 // This assumes all messages are PCP, we don't currently support NAT-PMP as
354 // we'd only need it on really old routers.
355
356 switch ( address->ss_family ) {
357 case AF_INET:
358 port = htons(((struct sockaddr_in *)address)->sin_port);
359 break;
360 case AF_INET6:
361 port = htons(((struct sockaddr_in6 *)address)->sin6_port);
362 break;
363 default:
364 GM_WARN_ONCE("PCP: decode_packet(): Address family %d.\n", address->ss_family);
365 return;
366 }
367
368 if ( port != PCP_SERVER_PORT ) {
369 GM_WARN_ONCE("PCP: Ignoring message that isn't from the PCP port.\n");
370 return;
371 }
372
373 switch ( address->ss_family ) {
374 case AF_INET:
375 if ( ((struct sockaddr_in *)address)->sin_addr.s_addr != GM.sta.ip4.router.sin_addr.s_addr ) {
376 GM_WARN_ONCE("IPV4 packet not from the router, ignored.\n");
377 return;
378 }
379 break;
380 case AF_INET6:
381 // FIX: Validate that IPV6 packet is from the router.
382 break;
383 }
384
385 response = p->opcode & 0x80;
386
387 switch ( p->opcode & 0x7f ) {
388 case PCP_ANNOUNCE:
389 decode_pcp_announce(p, message_size, address);
390 break;
391 case PCP_MAP:
392 case PCP_PEER:
393 if ( !response ) {
394 GM_WARN_ONCE("PCP client received request.\n");
395 return;
396 }
397 switch ( p->opcode & 0x7f ) {
398 case PCP_MAP:
399 if ( message_size < map_packet_size ) {
400 GM_WARN_ONCE("PCP receive packet too small for MAP: %d\n", message_size);
401 return;
402 }
403 decode_pcp_map(p, message_size, address);
404 break;
405 case PCP_PEER:
406 if ( message_size < sizeof(nat_pmp_or_pcp_t) ) {
407 GM_WARN_ONCE("PCP receive packet too small for PEER: %d\n", message_size);
408 return;
409 }
410 decode_pcp_peer(p, message_size, address);
411 break;
412 }
413 break;
414 default:
415 GM_WARN_ONCE("PCP unrecognized opcode %x\n", p->opcode);
416 }
417}

542gm_pcp_start_ipv4()
543{
544 // FIX: Time-out responses and retry, eventually abandon trying.
545 int yes = 1;
546 int no = 0;
547
548 if ( listener >= 0 )
549 return;
550
551 const struct sockaddr_in6 address = {
552 .sin6_family = AF_INET6,
553 // lwip defines struct in6_addr and associated things a bit differently than
554 // other platforms.
555 .sin6_addr = IN6ADDR_ANY_INIT,
556 .sin6_port = htons(PCP_CLIENT_PORT)
557 };
558
559 listener = socket(AF_INET6, SOCK_DGRAM, IPPROTO_UDP);
560
561 // Reuse addresses, because other software listens for all-hosts multicast.
562 // Reuse the port, because local_ipv6_socket also listens upon this port
563 // with a local address instead of the IPV6 ANY address.
564 setsockopt(listener, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(yes));
565 setsockopt(listener, SOL_SOCKET, SO_REUSEPORT, &yes, sizeof(yes));
566 // Accept both IPV4 and IPV6 connections.
567 (void) setsockopt(listener, IPPROTO_IPV6, IPV6_V6ONLY, (const void *)&no, sizeof(no));
568 if ( bind(listener, (struct sockaddr *)&address, sizeof(address)) < 0 ) {
569 GM_FAIL("bind failed.\n");
570 return;
571 }
572
573 struct ip_mreq m = {};
574 m.imr_interface.s_addr = GM.sta.ip4.address.sin_addr.s_addr;
575 inet_pton(AF_INET, "224.0.0.1", &m.imr_multiaddr.s_addr);
576 if ( setsockopt(listener, IPPROTO_IP, IP_ADD_MEMBERSHIP, &m, sizeof(m)) < 0 ) {
577 // This fails because the host is already registered to the "all-hosts" multicast
578 // group. Ignore that.
579 if ( errno != EADDRNOTAVAIL ) {
580 GM_FAIL("Setsockopt IP_ADD_MEMBERSHIP failed: %s.\n", strerror(errno));
581 return;
582 }
583 }
584 struct ipv6_mreq m6 = {};
585 inet_pton(AF_INET6, "ff02::1", &m6.ipv6mr_multiaddr.s6_addr);
586 m6.ipv6mr_interface = 0; // Default.
587 if ( setsockopt(listener, IPPROTO_IPV6, IPV6_JOIN_GROUP, &m6, sizeof(m6)) < 0 ) {
588 // This fails because the host is already registered to the "all-hosts" multicast
589 // group. Ignore that.
590 if ( errno != EADDRNOTAVAIL ) {
591 GM_FAIL("Setsockopt IP_ADD_MEMBERSHIP failed: %s.\n", strerror(errno));
592 return;
593 }
594 }
595 gm_fd_register(listener, incoming_packet, 0, true, false, true, 1);
596}

618 size_t this_bits = gm_match_bits(
619 &addresses[i],
620 &GM.sta.ip6.router.sin6_addr,
621 sizeof(*addresses));
622

626 }
627 }
628

631 return;
632 }
633

635 local_ipv6_address.sin6_port = htons(PCP_CLIENT_PORT);
636

639 GM_FAIL("Can't get a socket: %s\n", strerror(errno));
640 }
641
642 // Make the address reusable, because other things have bound to this address.
643 // Make the port reusable, because we listen to this port with a socket bound
644 // to the IPV6 ANY address, and with this socket.
645 int yes = 1;
646 setsockopt(local_ipv6_socket, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(yes));
647 setsockopt(local_ipv6_socket, SOL_SOCKET, SO_REUSEPORT, &yes, sizeof(yes));
648
649 if ( bind(local_ipv6_socket, (struct sockaddr *)&local_ipv6_address, sizeof(local_ipv6_address)) < 0 ) {
650 GM_FAIL("bind failed: %s.\n", strerror(errno));
651 return;
652 }
653 gm_fd_register(local_ipv6_socket, incoming_packet, 0, true, false, true, 0);
654}
