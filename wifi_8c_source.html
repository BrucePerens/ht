156static void
157ipv6_router_advertisement_handler(struct sockaddr_in6 * address, uint16_t router_lifetime)
158{
159 if ( !gm_all_zeroes(GM.sta.ip6.router.sin6_addr.s6_addr, sizeof(GM.sta.ip6.router.sin6_addr.s6_addr))
160 && memcmp(GM.sta.ip6.router.sin6_addr.s6_addr, address->sin6_addr.s6_addr, sizeof(address->sin6_addr.s6_addr)) != 0 ) {
161 GM_WARN_ONCE("Received a router advertisement from more than one IPv6 router. Ignoring all but the first.\n");
162 return;
163 }
164 memcpy(GM.sta.ip6.router.sin6_addr.s6_addr, address->sin6_addr.s6_addr, sizeof(address->sin6_addr.s6_addr));
165 GM.sta.ip6.router.sin6_family = AF_INET6;
166 GM.sta.ip6.router.sin6_port = 0;
167 if ( !pcp_ipv6_started ) {
168 pcp_ipv6_started = true;
169 gm_pcp_start_ipv6(&GM.sta);
170 gm_pcp_request_mapping_ipv6(&GM.sta);
171 }

184static void ip_event_sta_got_ip4(void* arg, esp_event_base_t event_base, int32_t event_id, void* event_data) {
185 ip_event_got_ip_t* event = (ip_event_got_ip_t*)event_data;
186 char buffer[INET6_ADDRSTRLEN + 1];
187
188 // Save the IP information for other facilities, like NAT-PCP, to use.
189 GM.sta.esp_netif = event->esp_netif;
190 GM.sta.ip4.address.sin_family = AF_INET;
191 GM.sta.ip4.address.sin_addr.s_addr = event->ip_info.ip.addr;
192 GM.sta.ip4.router.sin_family = AF_INET;
193 GM.sta.ip4.router.sin_addr.s_addr = event->ip_info.gw.addr;
194 GM.sta.ip4.netmask = event->ip_info.netmask.addr;
195
196 inet_ntop(AF_INET, &event->ip_info.ip.addr, buffer, sizeof(buffer));
197 gm_printf("Got IPv4: interface %s, address %s ", esp_netif_get_desc(event->esp_netif), buffer);
198 inet_ntop(AF_INET, &event->ip_info.gw.addr, buffer, sizeof(buffer));
199 gm_printf("router %s\n", buffer);
200 gm_sntp_start();
201 gm_stun(false, (struct sockaddr *)&GM.sta.ip4.router_public_ip, after_stun);
202 gm_pcp_start_ipv4(&GM.sta);
203 gm_pcp_request_mapping_ipv4(&GM.sta);
204 start_webserver();
205 gm_log_server_start();
206}

208// This handler is called when any netif gets an IPv6 address.
209static void ip_event_got_ip6(void* arg, esp_event_base_t event_base, int32_t event_id, void* event_data) {
210 const unsigned int public_address_count = \
211 sizeof(GM.sta.ip6.global) / sizeof(*GM.sta.ip6.global);
212 char buffer[INET6_ADDRSTRLEN + 1];
213
214 ip_event_got_ip6_t * event = (ip_event_got_ip6_t*)event_data;
215 esp_ip6_addr_type_t ipv6_type = esp_netif_ip6_get_addr_type(&event->ip6_info.ip);
216 const char * netif_name = esp_netif_get_desc(event->esp_netif);
217 gm_netif_t * interface;
218
219 inet_ntop(AF_INET6, &event->ip6_info.ip.addr, buffer, sizeof(buffer));
220 gm_printf("Got IPv6: interface %s, address %s, type %s\n",
221 netif_name,
222 buffer,
223 GM.ipv6_address_types[ipv6_type]);
224 fflush(stderr);
225
226 if (strcmp(netif_name, "sta") == 0)
227 interface = &GM.sta;
228 else if (strcmp(netif_name, "ap") == 0)
229 interface = &GM.ap;
230 else
231 return;
232
233 switch (ipv6_type) {
234 case ESP_IP6_ADDR_IS_UNKNOWN:
235 break;
236 case ESP_IP6_ADDR_IS_GLOBAL:
237 for ( unsigned int i = 0; i < public_address_count - 1; i++) {
238 // If we already have this address, don't add it twice.
239 if (memcmp(interface->ip6.global[i].sin6_addr.s6_addr, event->ip6_info.ip.addr, sizeof(event->ip6_info.ip.addr)) == 0)
240 break;
241 // If one of the public address entries is all zeroes, copy the address into it.
242 if (gm_all_zeroes(&interface->ip6.global[i].sin6_addr.s6_addr, sizeof(&interface->ip6.global[0].sin6_addr.s6_addr))) {
243 interface->ip6.global[i].sin6_family = AF_INET6;
244 memcpy(interface->ip6.global[i].sin6_addr.s6_addr, event->ip6_info.ip.addr, sizeof(event->ip6_info.ip.addr));
245 break;
246 }
247 }
248 break;
249 case ESP_IP6_ADDR_IS_LINK_LOCAL:
250 interface->ip6.link_local.sin6_family = AF_INET6;
251 memcpy(interface->ip6.link_local.sin6_addr.s6_addr, event->ip6_info.ip.addr, sizeof(event->ip6_info.ip.addr));
252 break;
253 case ESP_IP6_ADDR_IS_SITE_LOCAL:
254 interface->ip6.site_local.sin6_family = AF_INET6;
255 memcpy(interface->ip6.site_local.sin6_addr.s6_addr, event->ip6_info.ip.addr, sizeof(event->ip6_info.ip.addr));
256 break;
257 case ESP_IP6_ADDR_IS_UNIQUE_LOCAL:
258 interface->ip6.site_unique.sin6_family = AF_INET6;
259 memcpy(interface->ip6.site_unique.sin6_addr.s6_addr, event->ip6_info.ip.addr, sizeof(event->ip6_info.ip.addr));
260 break;
261 case ESP_IP6_ADDR_IS_IPV4_MAPPED_IPV6:
262 break;
263 }
264}

266static void wifi_connect_to_ap(const char * ssid, const char * password)
267{
268 wifi_config_t cfg;
269
270 bzero(&cfg, sizeof(cfg));
271 strncpy((char *)cfg.sta.ssid, ssid, sizeof(cfg.sta.ssid));
272 strncpy((char *)cfg.sta.password, password, sizeof(cfg.sta.password));
273
274 if (cfg.sta.password[0] == '\0')
275 cfg.sta.threshold.authmode = WIFI_AUTH_OPEN;
276 else
277 cfg.sta.threshold.authmode = WIFI_AUTH_WEP;
278
279 ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_STA));
280 ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_STA, &cfg));
281
282 ESP_ERROR_CHECK(esp_event_handler_instance_register(
283 WIFI_EVENT,
284 WIFI_EVENT_STA_CONNECTED,
285 &wifi_event_sta_connected_to_ap,
286 NULL,
287 &handler_wifi_event_sta_connected_to_ap));
288
289 ESP_ERROR_CHECK(esp_event_handler_instance_register(
290 IP_EVENT,
291 IP_EVENT_STA_GOT_IP,
292 &ip_event_sta_got_ip4,
293 NULL,
294 &handler_ip_event_sta_got_ip4));
295
296 ESP_ERROR_CHECK(esp_event_handler_instance_register(
297 IP_EVENT,
298 IP_EVENT_GOT_IP6,
299 &ip_event_got_ip6,
300 NULL,
301 &handler_ip_event_got_ip6));
302
303 ESP_ERROR_CHECK( esp_wifi_connect() );
304}
