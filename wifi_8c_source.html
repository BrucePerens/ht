233// This handler is called when any netif gets an IPv6 address.
234static void ip_event_got_ip6(void* arg, esp_event_base_t event_base, int32_t event_id, void* event_data) {
235 const unsigned int public_address_count = \
236 sizeof(GM.sta.ip6.global) / sizeof(*GM.sta.ip6.global);
237 char buffer[INET6_ADDRSTRLEN + 1];
238
239 ip_event_got_ip6_t * event = (ip_event_got_ip6_t*)event_data;
240 esp_ip6_addr_type_t ipv6_type = esp_netif_ip6_get_addr_type(&event->ip6_info.ip);
241 const char * netif_name = esp_netif_get_desc(event->esp_netif);
242 gm_netif_t * interface;
243 bool is_station = false;
244
245 inet_ntop(AF_INET6, &event->ip6_info.ip.addr, buffer, sizeof(buffer));
246 ; // gm_printf("Got IPv6: interface %s, address, type %s\n",
247 // netif_name,
248 // buffer,
249 // GM.ipv6_address_types[ipv6_type]);
250 fflush(stderr);
251
252 if (strcmp(netif_name, "sta") == 0) {
253 interface = &GM.sta;
254 is_station = true;
255 }
256 else if (strcmp(netif_name, "ap") == 0)
257 interface = &GM.ap;
258 else
259 return;
260
261 switch (ipv6_type) {
262 case ESP_IP6_ADDR_IS_UNKNOWN:
263 break;
264 case ESP_IP6_ADDR_IS_GLOBAL:
265 for ( unsigned int i = 0; i < public_address_count - 1; i++) {
266 // If we already have this address, don't add it twice.
267 if (memcmp(interface->ip6.global[i].sin6_addr.s6_addr, event->ip6_info.ip.addr, sizeof(event->ip6_info.ip.addr)) == 0)
268 break;
269 // If one of the public address entries is all zeroes, copy the address into it.
270 if (gm_all_zeroes(&interface->ip6.global[i].sin6_addr.s6_addr, sizeof(&interface->ip6.global[0].sin6_addr.s6_addr))) {
271 interface->ip6.global[i].sin6_family = AF_INET6;
272 memcpy(interface->ip6.global[i].sin6_addr.s6_addr, event->ip6_info.ip.addr, sizeof(event->ip6_info.ip.addr));
273 break;
274 }
275 }
276 break;
277 case ESP_IP6_ADDR_IS_LINK_LOCAL:
278 interface->ip6.link_local.sin6_family = AF_INET6;
279 memcpy(interface->ip6.link_local.sin6_addr.s6_addr, event->ip6_info.ip.addr, sizeof(event->ip6_info.ip.addr));
280 if (is_station) {
281 // FIX: We may never get a link-local address on some systems.
282 // Cope with it if we don't.
283 gm_port_control_protocol_start_listener_ipv6();
284 gm_port_control_protocol_request_mapping_ipv6();
285 gm_stun(true, (struct sockaddr *)&interface->ip6.pub, after_stun);
286 }
287 break;
288 case ESP_IP6_ADDR_IS_SITE_LOCAL:
289 interface->ip6.site_local.sin6_family = AF_INET6;
290 memcpy(interface->ip6.site_local.sin6_addr.s6_addr, event->ip6_info.ip.addr, sizeof(event->ip6_info.ip.addr));
291 break;
292 case ESP_IP6_ADDR_IS_UNIQUE_LOCAL:
293 interface->ip6.site_unique.sin6_family = AF_INET6;
294 memcpy(interface->ip6.site_unique.sin6_addr.s6_addr, event->ip6_info.ip.addr, sizeof(event->ip6_info.ip.addr));
295 break;
296 case ESP_IP6_ADDR_IS_IPV4_MAPPED_IPV6:
297 break;
298 }
299}

301static void sc_event_got_ssid_pswd(void* arg, esp_event_base_t event_base,
302 int32_t event_id, void* event_data)
303{
304 ESP_LOGD(TASK_NAME, "Got SSID and password");
305
306 smartconfig_event_got_ssid_pswd_t *evt = (smartconfig_event_got_ssid_pswd_t *)event_data;

308
309 bzero(&wifi_config, sizeof(wifi_config_t));
310 memcpy(wifi_config.sta.ssid, evt->ssid, sizeof(wifi_config.sta.ssid));

316
317 ESP_ERROR_CHECK(nvs_set_str(GM.nvs, "ssid", (const char *)evt->ssid));
318 ESP_ERROR_CHECK(nvs_set_str(GM.nvs, "wifi_password", (const char *)evt->password));
319 ESP_ERROR_CHECK(nvs_commit(GM.nvs));
320 ESP_ERROR_CHECK( esp_wifi_disconnect() );
321 ESP_ERROR_CHECK( esp_wifi_set_config(WIFI_IF_STA, &wifi_config) );
322 esp_wifi_connect();
323}
